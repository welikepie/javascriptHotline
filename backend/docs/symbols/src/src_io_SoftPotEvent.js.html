  6 JSUTILS.namespace('BO.io.SoftPotEvent');
  7 
  8 BO.io.SoftPotEvent = (function () {
  9 
 10     var SoftPotEvent;
 11 
 12     // Dependencies
 13     var Event = JSUTILS.Event;
 14 
 15     /**
 16      * @exports SoftPotEvent as BO.io.SoftPotEvent
 17      * @class An Event object to be dispatched (fired) by a SoftPot
 18      * object.   
 19      * @constructor
 20      * @augments JSUTILS.Event
 21      * @param {String} type The event type
 22      * @param {Number} touchPoint The value where the softpot was touched    
 23      */
 24     SoftPotEvent = function (type, touchPoint) {
 25 
 26         this.name = "SoftPotEvent";
 27 
 28         Event.call(this, type);
 29         this._touchPoint = touchPoint;
 30     };
 31 
 32     /** @constant */
 33     SoftPotEvent.PRESS = "softPotPressed";
 34     /** @constant */
 35     SoftPotEvent.RELEASE = "softPotRelease";
 36     /** @constant */
 37     SoftPotEvent.DRAG = "softPotDrag";
 38     /** @constant */
 39     SoftPotEvent.FLICK_UP = "softPotFlickUp";
 40     /** @constant */
 41     SoftPotEvent.FLICK_DOWN = "softPotFlickDown";
 42     /** @constant */
 43     SoftPotEvent.TAP = "softPotTap";        
 44 
 45     SoftPotEvent.prototype = JSUTILS.inherit(Event.prototype);
 46     SoftPotEvent.prototype.constructor = SoftPotEvent;
 47 
 48     /**
 49      * The value of the softpot.
 50      * 
 51      * @name SoftPotEvent#value
 52      * @property
 53      * @type Number
 54      */ 
 55     SoftPotEvent.prototype.__defineGetter__("value", function () { return this._touchPoint; });  
 56     SoftPotEvent.prototype.__defineSetter__("value", function (val) { this._touchPoint = val; });
 57 
 58     return SoftPotEvent;
 59 
 60 }());
