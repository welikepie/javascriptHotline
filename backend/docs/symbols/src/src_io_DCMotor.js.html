 12 JSUTILS.namespace('BO.io.DCMotor');
 13 
 14 BO.io.DCMotor = (function () {
 15 
 16     var DCMotor;
 17 
 18     // Dependencies
 19     var Pin = BO.Pin;
 20 
 21     /**
 22      * H-bridge motor control.
 23      * <p>Tested successfully with the following H-bridge: SN754410<br>
 24      * Should also be compatible with the following:<br>
 25      * SN754410<br>
 26      * L293NE<br>
 27      * TA7291P<br>
 28      * TB6612FNG<br>
 29      * BD621F</p>
 30      *
 31      * @exports DCMotor as BO.io.DCMotor
 32      * @class Creates an interface to an H-bridge to control the
 33      * direction of rotation of a motor shaft. You can rotate forward
 34      * (clockwise), reverse or apply a brake. See 
 35      * Breakout/examples/actuators/dcmotor.html for an example application.
 36      * @constructor
 37      * @param {IOBoard} board A reference to the IOBoard instance that the
 38      * servo is attached to.
 39      * @param {Pin} forwardPin A reference to the Pin connected to the forward
 40      * control of the H-bridge.
 41      * @param {Pin} reversePin A reference to the Pin connected to the reverse
 42      * control of the H-bridge.
 43      * @param {Pin} pwmPin A reference to the Pin connected to the pwm control
 44      * of the H-bridge.
 45      * @param {Number} minVoltage The minimum voltage (default = 1).
 46      * @param {Number} maxVoltage The maximum voltage (default = 9).
 47      * @param {Number} supplyVoltage The supply voltage (default = 9).
 48      */
 49     DCMotor = function (board, forwardPin, reversePin, pwmPin, minVoltage, maxVoltage, supplyVoltage) {
 50         "use strict";
 51         
 52         this.name = "DCMotor";
 53 
 54         minVoltage = minVoltage || 1;
 55         maxVoltage = maxVoltage || 9;
 56         supplyVoltage = supplyVoltage || 9;
 57         if (pwmPin === undefined) {
 58             pwmPin = null;
 59         }    
 60 
 61         this._value = 0;
 62         this._offset = 0;
 63         this._range = 0;
 64 
 65         this._forwardPin = forwardPin;
 66         this._reversePin = reversePin;
 67         this._pwmPin = pwmPin;
 68 
 69         if (this._pwmPin !== null) {
 70             if (this._pwmPin.getCapabilities()[Pin.PWM]) {
 71                 board.setDigitalPinMode(this._pwmPin.number, Pin.PWM);
 72             } else {
 73                 console.log("warning: PWM is not available for the PWM pin");
 74                 board.setDigitalPinMode(this._pwmPin.number, Pin.DOUT);
 75             }           
 76         }
 77 
 78         if (this._forwardPin.getCapabilities()[Pin.PWM]) {
 79             board.setDigitalPinMode(this._forwardPin.number, Pin.PWM);
 80         } else {
 81             console.log("warning: PWM is not available for the forward pin");
 82             board.setDigitalPinMode(this._forwardPin.number, Pin.DOUT);         
 83         }
 84 
 85         if (this._reversePin.getCapabilities()[Pin.PWM]) {
 86             board.setDigitalPinMode(this._reversePin.number, Pin.PWM);          
 87         } else {
 88             console.log("warning: PWM is not available for the reverse pin");
 89             board.setDigitalPinMode(this._reversePin.number, Pin.DOUT);
 90         }
 91 
 92         this._offset = (minVoltage / supplyVoltage);
 93         this._range = (maxVoltage - minVoltage) / supplyVoltage;
 94 
 95         this.despin(false);
 96 
 97     };
 98 
 99     DCMotor.prototype = {
100 
101         /**
102          * The value of the motor speed (-1.0 to 1.0). A speed of zero stops
103          * the motor.
104          * 
105          * @name DCMotor#value
106          * @property
107          * @type Number
108          */ 
109         set value(val) {
110             this._value = Math.max(-1, Math.min(1, val));
111 
112             if (val > 0) {
113                 this.forward(this._value);
114             } else if (val < 0) {
115                 this.reverse(-this._value);
116             } else {
117                 this.despin();
118             }
119         },
120         get value() {
121             return this._value;
122         },
123         
124         /**
125          * @param {Boolean} useBrake Default = true
126          */
127         despin: function (useBrake) {
128             if (useBrake === undefined) {
129                 useBrake = true;
130             }
131 
132             if (useBrake) {
133                 if (this._pwmPin === null) {
134                     this._forwardPin.value = 1;
135                     this._reversePin.value = 1;
136                 } else {
137                     this._forwardPin.value = 1;
138                     this._reversePin.value = 1;
139                     this._pwmPin.value = 1;
140                 }
141             } else {
142                 if (this._pwmPin === null) {
143                     this._forwardPin.value = 0;
144                     this._reversePin.value = 0;
145                 } else {
146                     this._forwardPin.value = 0;
147                     this._reversePin.value = 0;
148                     this._pwmPin.value = 0;
149                 }
150             }
151             this._value = 0;
152         },
153         
154         /**
155          * @param {Number} val The new voltage to set (0.0 to 1.0)
156          */     
157         forward: function (val) {
158             val = val || 1;
159             this._value = Math.max(0, Math.min(1, val));
160 
161             if (this._pwmPin === null) {
162                 this._forwardPin.value = Math.max(0, Math.min(1, this._value * this._range + this._offset));
163                 this._reversePin.value = 0;
164             } else {
165                 this._forwardPin.value = 1;
166                 this._reversePin.value = 0;
167                 this._pwmPin.value = Math.max(0, Math.min(1, this._value * this._range + this._offset));
168             }
169         },
170 
171         /**
172          * @param {Number} val The new voltage to set (-1.0 to 0.0)
173          */
174         reverse: function (val) {
175             val = val || 1;
176             this._value = Math.max(0, Math.min(1, val)) * -1;
177 
178             if (this._pwmPin === null) {
179                 this._forwardPin.value = 0;
180                 this._reversePin.value =  Math.max(0, Math.min(1, (this._value * this._range) * -1 + this._offset));
181             } else {
182                 this._forwardPin.value = 0;
183                 this._reversePin.value = 1;
184                 this._pwmPin.value = Math.max(0, Math.min(1, (this._value * this._range) * -1 + this._offset));
185             }
186         }       
187             
188     };
189 
190     return DCMotor;
191 
192 }());
