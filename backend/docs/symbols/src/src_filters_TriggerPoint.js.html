 12 JSUTILS.namespace('BO.filters.TriggerPoint');
 13 
 14 BO.filters.TriggerPoint = (function () {
 15     "use strict";
 16 
 17     var TriggerPoint;
 18 
 19     // dependencies
 20     var FilterBase = BO.filters.FilterBase;
 21 
 22     /**
 23      * Divides an input to 0 or 1 based on the threshold and hysteresis. You can also
 24      * use multiple points by providing a nested array such as [[0.4, 0.1], [0.7, 0.05]].
 25      *
 26      * @exports TriggerPoint as BO.filters.TriggerPoint
 27      * @class Divides an input to 0 or 1 based on the threshold and hysteresis. You can also
 28      * use multiple points by providing a nested array such as [[0.4, 0.1], [0.7, 0.05]].
 29      * See Breakout/examples/filters/triggerpoint.html for an example application.
 30      * @constructor
 31      * @augments BO.filters.FilterBase
 32      * @param {Number[]} points An array of threshold and hysteresis values
 33      * operations for input buffers.
 34      */
 35     TriggerPoint = function (points) {
 36 
 37         this.name = "TriggerPoint";
 38 
 39         this._points = {};
 40         this._range = [];
 41         this._lastStatus = 0;
 42 
 43         if (points === undefined) {
 44             points = [[0.5, 0]];
 45         }
 46 
 47         if (points[0] instanceof Array) {
 48             var len = points.length;
 49             for (var i = 0; i < len; i++) {
 50                 this._points[points[i][0]] = points[i][1];  
 51             }
 52         } else if (typeof points[0] === "number") {
 53             this._points[points[0]] = points[1];
 54         }
 55 
 56         this.updateRange();
 57 
 58         this._lastStatus = 0;
 59     };
 60 
 61 
 62     TriggerPoint.prototype = JSUTILS.inherit(FilterBase.prototype);
 63     TriggerPoint.prototype.constructor = TriggerPoint;
 64 
 65     /**
 66      * Override FilterBase.processSample
 67      *
 68      * @inheritDoc
 69      */
 70     TriggerPoint.prototype.processSample = function (val) {
 71         var status = this._lastStatus;
 72         var len = this._range.length;
 73         for (var i = 0; i < len; i++) {
 74             var range = this._range[i];
 75             if (range[0] <= val && val <= range[1]) {
 76                 status = i;
 77                 break;
 78             }
 79         }
 80 
 81         this._lastStatus = status;
 82         return status;
 83     };
 84 
 85     TriggerPoint.prototype.addPoint = function (threshold, hysteresis) {
 86         this._points[threshold] = hysteresis;
 87         this.updateRange();
 88     };
 89 
 90     TriggerPoint.prototype.removePoint = function (threshold) {
 91         // to do: verify that this works in javascript
 92         delete this._points[threshold];
 93         this.updateRange();
 94     };
 95 
 96     TriggerPoint.prototype.removeAllPoints = function () {
 97         this._points = {};
 98         this.updateRange();
 99     };
100 
101     /**
102      * @private
103      */
104     TriggerPoint.prototype.updateRange = function () {
105                 
106         this._range = [];
107         var keys = this.getKeys(this._points);
108 
109         var firstKey = keys[0];
110         this._range.push([Number.NEGATIVE_INFINITY, firstKey - this._points[firstKey]]);
111 
112         var len = keys.length - 1;
113         for (var i = 0; i < len; i++) {
114             var t0 = keys[i];
115             var t1 = keys[i + 1];
116             var p0 = (t0 * 1) + this._points[t0]; // multiply by 1 to force type to number
117             var p1 = t1 - this._points[t1];
118             if (p0 >= p1) {
119                 throw new Error("The specified range overlaps...");
120             }
121             this._range.push([p0, p1]);
122         }
123 
124         var lastKey = keys[keys.length - 1];
125         var positiveThresh = (lastKey * 1) + this._points[lastKey];
126         this._range.push([positiveThresh, Number.POSITIVE_INFINITY]);
127 
128     };
129 
130     /**
131      * @private
132      */
133     TriggerPoint.prototype.getKeys = function (obj) {
134         var keys = [];
135         for (var key in obj) {
136             if (obj.hasOwnProperty(key)) {
137                 keys.push(key);
138             }
139         }
140         return keys.sort();
141     };
142         
143     return TriggerPoint;
144 
145 }());
