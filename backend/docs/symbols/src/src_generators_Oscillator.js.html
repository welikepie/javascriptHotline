 12 JSUTILS.namespace('BO.generators.Oscillator');
 13 
 14 BO.generators.Oscillator = (function () {
 15     "use strict";
 16 
 17     var Oscillator;
 18 
 19     // dependencies
 20     var GeneratorBase = BO.generators.GeneratorBase,
 21         GeneratorEvent = BO.generators.GeneratorEvent,
 22         Timer = JSUTILS.Timer,
 23         TimerEvent = JSUTILS.TimerEvent;
 24 
 25     /**
 26      * Osc outputs a waveform on the associated PWM pin. For example, this can be used to blink or fade
 27      * an LED on or off.
 28      *
 29      * @exports Oscillator as BO.generators.Oscillator
 30      * @class The Oscillator object can be attached to a Pin or LED object to output
 31      * a waveform. This is useful for blinking an LED or fading it on and off. In
 32      * most cases (unless you are simply using it to blink and LED on or off), 
 33      * the Oscillator should be attached to a Pin or LED object associated with
 34      * a PWM pin on the I/O board. See Breakout/examples/generators/oscillator.html
 35      * for an example application.
 36      * @constructor
 37      * @augments BO.generators.GeneratorBase
 38      * @param {Number} wave waveform
 39      * @param {Number} freq frequency
 40      * @param {Number} amplitude amplitude
 41      * @param {Number} offset offset
 42      * @param {Number} phase phase
 43      * @param {Number} times The repeat count from 0 to infinite.
 44      */
 45     Oscillator = function (wave, freq, amplitude, offset, phase, times) {
 46 
 47         // call super class
 48         GeneratorBase.call(this);
 49 
 50         this.name = "Oscillator";
 51 
 52         this._wave = wave || Oscillator.SIN;
 53         this._freq = freq || 1;
 54         this._amplitude = amplitude || 1;
 55         this._offset = offset || 0;
 56         this._phase = phase || 0;
 57         this._times = times || 0;
 58 
 59         if (freq === 0) {
 60             throw new Error("Frequency should be larger than 0");
 61         }
 62 
 63         this._time = undefined;
 64         this._startTime = undefined;
 65         this._lastVal = undefined;
 66         // need to do this in order to remove the event listener
 67         this._autoUpdateCallback = this.autoUpdate.bind(this);
 68 
 69         this._timer = new Timer(33);
 70         this._timer.start();
 71 
 72         this.reset();
 73     };
 74 
 75     Oscillator.prototype = JSUTILS.inherit(GeneratorBase.prototype);
 76     Oscillator.prototype.constructor = Oscillator;
 77 
 78     /**
 79      * The service interval in milliseconds. Default is 33ms.
 80      *
 81      * @name Oscillator#serviceInterval
 82      * @property
 83      * @type Number
 84      */ 
 85     Oscillator.prototype.__defineSetter__("serviceInterval", function (interval) {
 86         this._timer.delay = interval;
 87     });
 88     Oscillator.prototype.__defineGetter__("serviceInterval", function () {
 89         return this._timer.delay;
 90     });
 91 
 92     /**
 93      * Starts the oscillator
 94      */
 95     Oscillator.prototype.start = function () {
 96         this.stop();
 97         this._timer.addEventListener(TimerEvent.TIMER, this._autoUpdateCallback);
 98 
 99         var date = new Date();
100         this._startTime = date.getTime();
101         this.autoUpdate(null);
102     };
103 
104     /**
105      * Stops the oscillator.
106      */
107     Oscillator.prototype.stop = function () {
108         if (this._timer.hasEventListener(TimerEvent.TIMER)) {
109             this._timer.removeEventListener(TimerEvent.TIMER, this._autoUpdateCallback);
110         }
111     };
112 
113     /**
114      * Resets the oscillator.
115      */
116     Oscillator.prototype.reset = function () {
117         this._time = 0;
118         this._lastVal = 0.999;
119     };
120 
121     /**
122      * By default the interval is 33 milliseconds. The Osc is updated every 33ms.
123      * @param {Number} interval The update interval in milliseconds.
124      */
125     Oscillator.prototype.update = function (interval) {
126         interval = interval || -1;
127         if (interval < 0) {
128             this._time += this._timer.delay;
129         }
130         else {
131             this._time += interval;
132         }
133         this.computeValue();
134     };
135 
136     /**
137      * @private
138      */
139     Oscillator.prototype.autoUpdate = function (event) {
140         var date = new Date();
141         this._time = date.getTime() - this._startTime;
142         this.computeValue();
143     };
144 
145     /**
146      * @private
147      */
148     Oscillator.prototype.computeValue = function () {
149         var sec = this._time / 1000;
150 
151         if (this._times !== 0 && this._freq * sec >= this._times) {
152             this.stop();
153             sec = this._times / this._freq;
154             if (this._wave !== Oscillator.LINEAR) {
155                 this._value = this._offset;
156             } else {
157                 this._value = this._amplitude * this._wave(1, 0) + this._offset;
158             }
159         } else {
160             var val = this._freq * (sec + this._phase);
161             this._value = this._amplitude * this._wave(val, this._lastVal) + this._offset;
162             this._lastVal = val;
163         }
164         this.dispatchEvent(new GeneratorEvent(GeneratorEvent.UPDATE));
165     };
166 
167     // Static methods
168 
169     /**
170      * sine wave
171      * @static
172      */
173     Oscillator.SIN = function (val, lastVal) {
174         return 0.5 * (1 + Math.sin(2 * Math.PI * (val - 0.25)));
175     };
176 
177     /**
178      * square wave
179      * @static
180      */
181     Oscillator.SQUARE = function (val, lastVal) {
182         return (val % 1 <= 0.5) ? 1 : 0;
183     };
184     
185     /**
186      * triangle wave
187      * @static
188      */
189     Oscillator.TRIANGLE = function (val, lastVal) {
190         val %= 1;
191         return (val <= 0.5) ? (2 * val) : (2 - 2 * val);
192     };
193     
194     /**
195      * saw wave
196      * @static
197      */
198     Oscillator.SAW = function (val, lastVal) {
199         val %= 1;
200         if (val <= 0.5) {
201             return val + 0.5;
202         }
203         else {
204             return val - 0.5;
205         }
206     };
207     
208     /**
209      * impulse
210      * @static
211      */
212     Oscillator.IMPULSE = function (val, lastVal) {
213         return ((val % 1) < (lastVal % 1)) ? 1 : 0;
214     };
215     
216     /**
217      * linear
218      * @static
219      */
220     Oscillator.LINEAR = function (val, lastVal) {
221         return (val < 1) ? val : 1;
222     };
223     
224     // document events
225 
226     /**
227      * The update event is dispatched at the rate specified 
228      * by the serviceInterval parameter (default = 33ms).
229      * @name Oscillator#update
230      * @type BO.generators.GeneratorEvent.UPDATE
231      * @event
232      * @param {BO.generators.Oscillator} target A reference to the Oscillator object.
233      */
234 
235     return Oscillator;
236 
237 
238 }());
