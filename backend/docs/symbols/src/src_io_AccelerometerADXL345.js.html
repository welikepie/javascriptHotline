 11     var AccelerometerADXL345;
 12 
 13     // private static constants
 14     var RAD_TO_DEG = 180 / Math.PI,
 15         POWER_CTL = 0x2D,
 16         DATAX0 = 0x32,
 17         DATA_FORMAT = 0x31,
 18         OFSX = 0x1E,
 19         OFSY = 0x1F,
 20         OFSZ = 0x20,
 21         ALL_AXIS =  DATAX0 | 0x80,
 22         NUM_BYTES = 6;  
 23 
 24     // dependencies
 25     var I2CBase = BO.I2CBase,
 26         AccelerometerEvent = BO.io.AccelerometerEvent;

 28     /**
 29      * Analog Devices ADXL345 3-axis accelerometer
 30      *
 31      * @exports AccelerometerADXL345 as BO.io.AccelerometerADXL345
 32      * @class Creates an interface to an ADXL345 3-axis accelerometer. Use the
 33      * accelerometer to read the acceleration along the x, y, and z axis of an 
 34      * object it is attached to. You can also obtain the pitch and roll. See the
 35      * example in Breakout/examples/sensors/adxl345.html.
 36      * @constructor
 37      * @augments BO.I2CBase
 38      * @param {IOBoard} board The IOBoard instance
 39      * @param {Number} range The dynamic range selection in Gs (options RANGE_2G, RANGE_4G, 
 40      * RANGE_8G, RANGE_16G). Default is RANGE_2G.    
 41      * @param {Number} address The i2c address of the accelerometer (default is 0x53)
 42      */
 43     AccelerometerADXL345 = function (board, range, address) {
 44 
 45         address = address || AccelerometerADXL345.DEVICE_ID;
 46         I2CBase.call(this, board, address);
 47 
 48         this.name = "AccelerometerADXL345";
 49 
 50         this._dynamicRange = range || AccelerometerADXL345.RANGE_2G;
 51         
 52         this._sensitivity = {
 53             x: AccelerometerADXL345.DEFAULT_SENSITIVITY,
 54             y: AccelerometerADXL345.DEFAULT_SENSITIVITY,
 55             z: AccelerometerADXL345.DEFAULT_SENSITIVITY,
 56         };
 57 
 58         this._offset = {x: 0, y: 0, z: 0};
 59 
 60         this._isReading = false;
 61         this._debugMode = BO.enableDebugging;
 62 
 63         this._x = 0;
 64         this._y = 0;
 65         this._z = 0;
 66         this._rawX = 0;
 67         this._rawY = 0;
 68         this._rawZ = 0;
 69 
 70         // initiate the device
 71         this.powerOn();
 72 
 73         // sets the dynamic range and sets teh full_res bit
 74         this.setRangeAndFullRes(this._dynamicRange);
 75 
 76     };
 77

 79     AccelerometerADXL345.prototype.constructor = AccelerometerADXL345;

123         // -180 to 180

137         //return Math.atan2(this._y, this._z) * RAD_TO_DEG;
138         // -90 to 90
139         return Math.atan2(this._y, Math.sqrt(this._x * this._x + this._z * this._z)) * RAD_TO_DEG;

204     /**
205      * @private
206      */
207     AccelerometerADXL345.prototype.setRangeAndFullRes = function (range) {
208             
209         var setting;
210         
211         switch (range) {
212         case 2:
213             setting = 0x00;
214             break;
215         case 4:
216             setting = 0x01;
217             break;
218         case 8:
219             setting = 0x02;
220             break;
221         case 16:
222             setting = 0x03;
223             break;
224         default:
225             setting = 0x00;
226             break;
227         }
228         
229         // set full scale bit (3) and range bits (0 - 1)
230         setting |= (0x08 & 0xEC);
231         this.sendI2CRequest([I2CBase.WRITE, this._address, DATA_FORMAT, setting]);
232     };

234     /**
235      * @private
236      */
237     AccelerometerADXL345.prototype.handleI2C = function (data) {
238         switch (data[0]) {
239         case ALL_AXIS:
240             this.readAccel(data);
241             break;
242         case OFSX:
243             this.debug("offset x = " + data[2]);
244             break;
245         case OFSY:
246             this.debug("offset y = " + data[2]);
247             break;
248         case OFSZ:
249             this.debug("offset z = " + data[2]);
250             break;
251         }
252     };

254     /**
255      * Start continuous reading of the sensor
256      */
257     AccelerometerADXL345.prototype.startReading = function () {
258         if (!this._isReading) {
259             this._isReading = true;
260             this.sendI2CRequest([I2CBase.READ_CONTINUOUS, this.address, ALL_AXIS, NUM_BYTES]);
261         }
262     };

264     /**
265      * Stop continuous reading of the sensor
266      */
267     AccelerometerADXL345.prototype.stopReading = function () {
268         this._isReading = false;
269         this.sendI2CRequest([I2CBase.STOP_READING, this.address]);
270     };

272     /**
273      * offset the x, y, or z axis output by the respective input value
274      */     
275     AccelerometerADXL345.prototype.setAxisOffset = function (xVal, yVal, zVal) {
276         // store values so we can retrieve via getAxisOffset
277         this._offset.x = xVal;
278         this._offset.y = yVal;
279         this._offset.z = zVal;
280         
281         this.sendI2CRequest([I2CBase.WRITE, this.address, OFSX, xVal]);
282         this.sendI2CRequest([I2CBase.WRITE, this.address, OFSY, yVal]);
283         this.sendI2CRequest([I2CBase.WRITE, this.address, OFSZ, zVal]);
284     };

286     /**
287      * get the value of the x, y, and z axis offset
288      */
289     AccelerometerADXL345.prototype.getAxisOffset = function () {
290         // will trace values if debug mode is enabled
291         this.sendI2CRequest([I2CBase.READ, this.address, OFSX, 1]);
292         this.sendI2CRequest([I2CBase.READ, this.address, OFSY, 1]);
293         this.sendI2CRequest([I2CBase.READ, this.address, OFSZ, 1]);
294         
295         // return the locally stored values because it is not possible
296         // without a more elaborate design to get i2c read values
297         // in a single call
298         return this._offset;
299     };

301     /** 
302      * Sends read request to accelerometer and updates accelerometer values.
303      */
304     AccelerometerADXL345.prototype.update = function () {
305         if (this._isReading) {
306             this.stopReading(); 
307         }
308         // read data: contents of X, Y, and Z registers
309         this.sendI2CRequest([I2CBase.READ, this.address, ALL_AXIS, NUM_BYTES]);
310     };
311 
312     /**
313      * @private
314      */
315     AccelerometerADXL345.prototype.powerOn = function () {
316 
317         // standby mode
318         this.sendI2CRequest([I2CBase.WRITE, this.address, POWER_CTL, 0]);
319         
320         // set measure bit
321         this.setRegisterBit(POWER_CTL, 3, true);
322     };
323     
324     /**
325      * @private
326      */
327     AccelerometerADXL345.prototype.setRegisterBit = function (regAddress, bitPos, state) {
328         var value;
329         
330         if (state) {
331             value |= (1 << bitPos);
332         } else {
333             value &= ~(1 << bitPos);
334         }
335 
336         this.sendI2CRequest([I2CBase.WRITE, this.address, regAddress, value]);
337     };
338 
339     /**
340      * @private
341      */
342     AccelerometerADXL345.prototype.readAccel = function (data) {
343         
344         var x_val,
345             y_val,
346             z_val;
347 
348         if (data.length != NUM_BYTES + 1) {
349             throw new Error("Incorrect number of bytes returned");
350         }
351         
352         x_val = (data[2] << 8) | (data[1]);
353         y_val = (data[4] << 8) | (data[3]);
354         z_val = (data[6] << 8) | (data[5]);
355         
356         if (x_val >> 15) {
357             this._rawX = ((x_val ^ 0xFFFF) + 1) * -1;
358         } else {
359             this._rawX = x_val;
360         }
361         if (y_val >> 15) {
362             this._rawY = ((y_val ^ 0xFFFF) + 1) * -1;
363         } else {
364             this._rawY = y_val;
365         }
366         if (z_val >> 15) {
367             this._rawZ = ((z_val ^ 0xFFFF) + 1) * -1;
368         } else {
369             this._rawZ = z_val;
370         }
371 
372         this._x = this._rawX * this._sensitivity.x;
373         this._y = this._rawY * this._sensitivity.y;
374         this._z = this._rawZ * this._sensitivity.z;
375         
376         this.dispatchEvent(new AccelerometerEvent(AccelerometerEvent.UPDATE));          
377     };
