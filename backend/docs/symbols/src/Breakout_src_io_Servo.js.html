 12 JSUTILS.namespace('BO.io.Servo');
 13 
 14 BO.io.Servo = (function() {
 15 
 16 	var Servo;
 17 
 18 	// dependencies
 19 	var Pin = BO.Pin;
 20 
 21 	/**
 22 	 * Creates a new Servo.
 23 	 *
 24 	 * @exports Servo as BO.io.Servo
 25 	 * @class Creates an interface to a Servo motor. Use this object to set
 26 	 * the angle of the servo head. You can simply specify and angle between
 27 	 * 0 and 180 degrees and the servo head will rotate to that angle. See
 28 	 * Breakout/examples/actuators/servo.html for an example application.
 29 	 * @constructor
 30 	 * @param {IOBoard} board A reference to the IOBoard instance that the 
 31 	 * servo is attached to.
 32 	 * @param {Pin} servoPin A reference to the Pin the servo is connected to.
 33 	 * @param {Number} minAngle The minimum angle the server can rotate to
 34 	 * (default = 0).
 35 	 * @param {Number} maxAngle The maximum angle the server can rotate to
 36 	 * (default = 180).
 37 	 */
 38 	Servo = function(board, servoPin, minAngle, maxAngle) {
 39 		"use strict";
 40 		
 41 		this.name = "Servo";
 42 
 43 		this._pin = servoPin;
 44 		this._angle;
 45 		this._minAngle = minAngle || 0;
 46 		this._maxAngle = maxAngle || 180;
 47 
 48 		var pinNumber = servoPin.number;
 49 		
 50 		// sendServoAttach will set the pin mode to Pin.SERVO
 51 		board.sendServoAttach(pinNumber);
 52 	};
 53 
 54 	Servo.prototype = {
 55 
 56 		/**
 57 		 * Set the angle (in degrees) to rotate the server head to.
 58 		 * 
 59 		 * @name Servo#angle
 60 		 * @property
 61 		 * @type Number
 62 		 */ 
 63 		set angle(value) {
 64 			if (this._pin.getType() == Pin.SERVO) {
 65 				this._angle = value;
 66 				//this._pin.value = this._angle;
 67 				this._pin.value = Math.max(0, Math.min(1, (this._angle - this._minAngle) / 
 68 								(this._maxAngle - this._minAngle) * Servo.COEF_TO_0_180));
 69 
 70 			}
 71 		},
 72 		get angle() {
 73 			if (this._pin.getType() == Pin.SERVO) {
 74 				return this._angle;
 75 			}
 76 		}		
 77 	};
 78 
 79 	/**
 80 	 * The scale to convert 0-1 (0-255 in 8bit) to 0-0.706 (0-180 in 8bit).
 81 	 */
 82 	Servo.COEF_TO_0_180 = 180 / 255;
 83 
 84 	return Servo;
 85 
 86 }());
