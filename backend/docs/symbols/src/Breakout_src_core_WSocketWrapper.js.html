  6 JSUTILS.namespace('BO.WSocketWrapper');
  7 
  8 BO.WSocketWrapper = (function() {
  9 	"use strict";
 10 
 11  	var WSocketWrapper;
 12 
 13  	// dependencies
 14  	var EventDispatcher = JSUTILS.EventDispatcher,
 15  		WSocketEvent = BO.WSocketEvent;
 16 
 17 	/**
 18 	 * Creates a wrapper for various websocket implementations to unify the interface.
 19 	 *
 20 	 * @exports WSocketWrapper as BO.WSocketWrapper
 21 	 * @class Creates a wrapper for various websocket implementations to unify the interface.
 22 	 * @constructor
 23 	 * @param {String} host The host address of the web server.
 24 	 * @param {Number} port The port to connect to on the web server.
 25 	 * @param {Boolean} useSocketIO Set true to use socket.io implementation, set false to use
 26 	 * native websocket implementation.
 27 	 * @param {String} protocol The websockt protocol definition (if necessary).
 28 	 */
 29 	WSocketWrapper = function(host, port, useSocketIO, protocol) {
 30 		this.name = "WSocketWrapper";
 31 
 32 		EventDispatcher.call(this, this);
 33 
 34 		this._host = host;
 35 		this._port = port;
 36 		this._protocol = protocol || "default-protocol";
 37 		this._useSocketIO = useSocketIO || false;
 38 		this._socket = null;
 39 		this._readyState = ""; // only applies to native WebSocket implementations
 40 
 41 		this.init(this);
 42 
 43 	};
 44 
 45 	WSocketWrapper.prototype = JSUTILS.inherit(EventDispatcher.prototype);
 46 	WSocketWrapper.prototype.constructor = WSocketWrapper;
 47 
 48 	/**
 49 	 * Initialize the websocket
 50 	 * @param {Object} self A reference to this websocket object.
 51 	 * @private
 52 	 */
 53 	WSocketWrapper.prototype.init = function(self) {
 54 
 55 		if (self._useSocketIO) {
 56 			self._socket = io.connect("http://"+self._host+":"+self._port);
 57 
 58 			try {
 59 				/** @private */
 60 				self._socket.on('connect', function() {
 61 					self.dispatchEvent(new WSocketEvent(WSocketEvent.CONNECTED));
 62 					/** @private */
 63 					self._socket.on('message', function(msg) {
 64 						self.dispatchEvent(new WSocketEvent(WSocketEvent.MESSAGE), {message: msg});
 65 					});
 66 				});
 67 				
 68 			} catch(exception){
 69 				console.log("Error "+exception);
 70 			}
 71 
 72 		} else {
 73 			
 74 			try {
 75 
 76 				if ("MozWebSocket" in window) {
 77 					// MozWebSocket is no longer used in Firefox 11 and higher
 78 					self._socket = new MozWebSocket("ws://"+self._host+":"+self._port, self._protocol);
 79 				} else if ("WebSocket" in window) {
 80 					// Safari doesn't like protocol parameter
 81 					//self._socket = new WebSocket("ws://"+self._host+":"+self._port, self._protocol);
 82 					self._socket = new WebSocket("ws://"+self._host+":"+self._port);
 83 				} else {
 84 					console.log("Websockets not supported by this browser");
 85 					throw "Websockets not supported by this browser";
 86 				}
 87 				/** @private */
 88 				self._socket.onopen = function() {
 89 
 90 					self.dispatchEvent(new WSocketEvent(WSocketEvent.CONNECTED));
 91 					/** @private */
 92 					self._socket.onmessage = function(msg) {
 93 						self.dispatchEvent(new WSocketEvent(WSocketEvent.MESSAGE), {message: msg.data});
 94 					};
 95 					/** @private */
 96 					self._socket.onclose = function() {
 97 						self._readyState = self._socket.readyState;
 98 						self.dispatchEvent(new WSocketEvent(WSocketEvent.CLOSE));	
 99 					};
100 
101 				};
102 				
103 			} catch(exception){
104 				console.log("Error "+exception);
105 			}
106 				
107 		}
108 
109 	};
110 
111 	/**
112 	 * Send a message
113 	 * TO DO: support sending ArrayBuffers and Blobs
114 	 * for now, forward any calls to sendString
115 	 * @private
116 	 * @param {String} message The message to send
117 	 */
118 	WSocketWrapper.prototype.send = function(message) {
119 		// to do: ensure socket is not null before trying to send
120 		//this._socket.send();
121 		this.sendString(message);
122 	};
123 
124 	/**
125 	 * Send a message
126 	 * @param {String} message The message to send
127 	 */
128 	WSocketWrapper.prototype.sendString = function(message) {
129 		// to do: ensure socket is not null before trying to send
130 		this._socket.send(message.toString());
131 	};	
132 
133 	// to do: ensure socket is not null before trying to get readyState
134 
135 	/**
136 	 * [read-only] Wrapper for the readyState method of the native websocket implementation
137 	 * <p>CONNECTING = 0, OPEN = 1, CLOSING = 2, CLOSED = 3</p>
138 	 * @name WSocketWrapper#readyState
139 	 * @property
140 	 * @type String
141 	 */		 
142 	WSocketWrapper.prototype.__defineGetter__("readyState", function() { return this._readyState; });
143 
144 
145 	// document events
146 
147 	/**
148 	 * The webSocketConnected event is dispatched when a connection with
149 	 * the websocket is established.
150 	 * @name WSocketWrapper#webSocketConnected
151 	 * @type BO.WebsocketEvent.CONNECTED
152 	 * @event
153 	 * @param {BO.WSocketWrapper} target A reference to the WSocketWrapper object.
154 	 */	
155 
156 	/**
157 	 * The webSocketMessage event is dispatched when a websocket message is received.
158 	 * @name WSocketWrapper#webSocketMessage
159 	 * @type BO.WebsocketEvent.MESSAGE
160 	 * @event
161 	 * @param {BO.WSocketWrapper} target A reference to the WSocketWrapper object.
162 	 * @param {String} message The websocket data	 
163 	 */	
164 
165 	/**
166 	 * The webSocketClosed event is dispatched the websocket connection is closed.
167 	 * @name WSocketWrapper#webSocketClosed
168 	 * @type BO.WebsocketEvent.CLOSE
169 	 * @event
170 	 * @param {BO.WSocketWrapper} target A reference to the WSocketWrapper object. 
171 	 */		 
172 
173 	return WSocketWrapper;
174 
175 }());
