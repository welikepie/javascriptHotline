  6 JSUTILS.namespace('BO.IOBoardEvent');
  7 
  8 BO.IOBoardEvent = (function() {
  9 
 10 	var IOBoardEvent;
 11 
 12 	// Dependencies
 13 	var Event = JSUTILS.Event;
 14 
 15 	/**
 16 	 * @exports IOBoardEvent as BO.IOBoardEvent
 17 	 * @class An Event object to be dispatched (fired) by the IOBoard object.
 18 	 * The most important event is the READY event which signifies that the
 19 	 * I/O board is ready to receive commands from the application. Many of the
 20 	 * other IOBoard events are used when creating new io component objects.
 21 	 * @constructor
 22 	 * @augments JSUTILS.Event
 23 	 * @param {String} type The event type
 24 	 */
 25 	IOBoardEvent = function(type) {
 26 
 27 		this.name = "IOBoardEvent";
 28 		
 29 		// Call the super class
 30 		// 2nd parameter is passed to EventDispatcher constructor
 31 		Event.call(this, type);
 32 	};
 33 
 34 	// Events
 35 	/** @constant */
 36 	IOBoardEvent.ANALOG_DATA = "analogData";
 37 	/** @constant */
 38 	IOBoardEvent.DIGITAL_DATA = "digitalData";
 39 	/** @constant */
 40 	IOBoardEvent.FIRMWARE_VERSION = "firmwareVersion";
 41 	/** @constant */
 42 	IOBoardEvent.FIRMWARE_NAME = "firmwareName";
 43 	/** @constant */
 44 	IOBoardEvent.STRING_MESSAGE = "stringMessage";
 45 	/** @constant */
 46 	IOBoardEvent.SYSEX_MESSAGE = "sysexMessage";
 47 	/** @constant */
 48 	IOBoardEvent.PIN_STATE_RESPONSE = "pinStateResponse";
 49 	/** @constant */
 50 	IOBoardEvent.READY = "ioBoardReady";
 51 	/** @constant */
 52 	IOBoardEvent.CONNECTED = "ioBoardConnected";
 53 	/** @constant */
 54 	IOBoardEvent.DISCONNECTED = "ioBoardDisonnected";		
 55 
 56 	IOBoardEvent.prototype = JSUTILS.inherit(Event.prototype);
 57 	IOBoardEvent.prototype.constructor = IOBoardEvent;
 58 
 59 	return IOBoardEvent;
 60 
 61 }());
