1 /**
  2  * This creates a namespace for common javascript utility objects and also
  3  * contains a few utility functions.
  4  *
  5  * inherit() method copied from Flanagan, David. JavaScript: The Definitive Guide.
  6  */
  7 
  8 /** @namespace Namespace and utility functions */
  9 var JSUTILS = JSUTILS || {};
 10 
 11 
 12 // Utility functions
 13 
 14 /** 
 15  * Use this function to safely create a new namespace
 16  * if a namespace already exists, it won't be recreated.
 17  *
 18  * @function
 19  * @param {String} namespaceString The namespace as a string.
 20  */
 21 JSUTILS.namespace = function (namespaceString) {
 22 	var parts = namespaceString.split('.'),
 23 		parent = window,
 24 		i;
 25 			
 26 	for (i=0; i<parts.length; i +=1) {
 27 		// create a property if it doesn't exist
 28 		if (typeof parent[parts[i]] === "undefined") {
 29 			parent[parts[i]] = {};
 30 		}
 31 		parent = parent[parts[i]];
 32 	}
 33 	return parent;
 34 };
 35 
 36 /**
 37  * Use this method rather than Object.create() directly if
 38  * browser compatibility is unknown.
 39  *
 40  * @function
 41  * @param {Object} p The prototype of the object to inherit.
 42  */
 43 JSUTILS.inherit = function(p) {
 44 	if (p == null) throw TypeError(); // p must be a non-null object
 45 	if (Object.create) { // If Object.create() is defined...
 46 		return Object.create(p); // then just use it
 47 	}
 48 	var t = typeof p; // otherwise do some more type checking
 49 	if (t !== "object" && t !== "function") throw TypeError();
 50 	function f() {}; // define a dummy constructor function
 51 	f.prototype = p; // Set its prototype property to p
 52 	return new f(); // use f() to create an 'heir' of p.
 53 };

 56 // Copied from https://developer.mozilla.org/en/JavaScript/Reference/Global_Objects/Function/bind
 57 if (!Function.prototype.bind) {  
 58 
 59 	/** 
 60 	 * add bind for browsers that don't support it (Safari)
 61 	 * @function
 62 	 * @private
 63 	 */
 64   	Function.prototype.bind = function (oThis) {  
 65     if (typeof this !== "function") {  
 66       // closest thing possible to the ECMAScript 5 internal IsCallable function  
 67       throw new TypeError("Function.prototype.bind - what is trying to be bound is not callable");  
 68     }  
 69   
 70     var aArgs = Array.prototype.slice.call(arguments, 1),   
 71         fToBind = this, 
 72         /** 
 73          * @private
 74          */  
 75         fNOP = function () {},
 76         /** 
 77          * @private
 78          */  
 79         fBound = function () {  
 80           return fToBind.apply(this instanceof fNOP  
 81                                  ? this  
 82                                  : oThis || window,  
 83                                aArgs.concat(Array.prototype.slice.call(arguments)));  
 84         };  
 85   
 86     fNOP.prototype = this.prototype;  
 87     fBound.prototype = new fNOP();  
 88   
 89     return fBound;  
 90   };  
 91 }
