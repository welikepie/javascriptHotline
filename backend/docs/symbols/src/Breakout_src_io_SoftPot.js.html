  8 BO.io.SoftPot = (function() {
  9 
 10 	var SoftPot;
 11 
 12 	// private static constants:
 13 	var	TAP_TIMEOUT				= 200,
 14 		FLICK_TIMEOUT			= 200,
 15 		PRESS_TIMER_INTERVAL	= 10,
 16 		MIN_VALUE				= 0.01,
 17 		DEBOUNCE_TIMEOUT		= 20;	
 18 
 19 	// dependencies
 20 	var PhysicalInputBase = BO.PhysicalInputBase,
 21 		Pin = BO.Pin,
 22 		PinEvent = BO.PinEvent,
 23 		Scaler = BO.filters.Scaler,
 24 		Timer = JSUTILS.Timer,
 25 		TimerEvent = JSUTILS.TimerEvent,
 26 		SoftPotEvent = BO.io.SoftPotEvent;

 28 	/**
 29 	 * A softpot analog sensor.
 30 	 *
 31 	 * @exports SoftPot as BO.io.SoftPot
 32 	 * @class Creates an interface to a SoftPot sensor. A softpot is a type of
 33 	 * analog resistive sensor that acts as a type of slider input. There are 
 34 	 * straight and curved variations. This object provides a number of useful 
 35 	 * events such as Press, Release, Drag, Tap and capturing Flick gestures.
 36 	 * See Breakout/examples/sensors/softpot.html for an example application.
 37 	 * @constructor
 38 	 * @augments BO.PhysicalInputBase
 39 	 * @param {IOBoard} board A reference to the IOBoard instance
 40 	 * @param {Pin} pin A reference to the Pin the softpot is connected to.
 41 	 * @param {Number} softPotLength The length of the softpot in mm. 
 42 	 * Default = 100. 
 43 	 */
 44 	SoftPot = function(board, pin, softPotLength) {
 45 		"use strict";
 46 		
 47 		PhysicalInputBase.call(this);
 48 
 49 		this.name = "SoftPot";
 50 		this._pin = pin;
 51 
 52 		softPotLength = softPotLength || 100;
 53 
 54 		this._isDrag = false;
 55 		this._flickDistance = 0;
 56 		this._touchPoint = 0;
 57 		this._lastMovePoint = 0;
 58 		this._isTouched = false;
 59 		this._distanceFromPressed = 0;
 60 		this._minFlickMovement = 1.0/softPotLength * 2.5;
 61 		this._minDragMovement = 1.0/softPotLength * 1.0;
 62 		this._tapTimeout = TAP_TIMEOUT;
 63 		this._minValue = MIN_VALUE;
 64 		
 65 		this._board = board;
 66 		board.enableAnalogPin(this._pin.analogNumber);
 67 
 68 		this._debugMode = BO.enableDebugging;
 69 						
 70 		this._pin.addEventListener(PinEvent.CHANGE, this.onPinChange.bind(this));
 71 
 72 		this._pressTimer = new Timer(PRESS_TIMER_INTERVAL, 0);
 73 		this._flickTimer = new Timer(FLICK_TIMEOUT, 1);
 74 	};
 75 
 76 
 77 	SoftPot.prototype = JSUTILS.inherit(PhysicalInputBase.prototype);
 78 	SoftPot.prototype.constructor = SoftPot;
 79 
 80 	/**
 81 	 * @private
 82 	 * @param {Event} evt PinEvent.CHANGE
 83 	 */
 84 	SoftPot.prototype.onPinChange = function(evt) {
 85 		var val = evt.target.value;
 86 
 87 		// _minValue is the minimum value required to set the release state
 88 		// should be as close to zero as possible
 89 		if (this._isTouched && val < this._minValue) {
 90 			this.onRelease();
 91 		} else if (val >= this._minValue) {
 92 			if (!this._isTouched) {
 93 				this.startTouch(val);
 94 				this._lastMovePoint = val;
 95 			} else {
 96 				this.onMove(val);
 97 			}
 98 		}
 99 	};
100 
101 	/**
102 	 * @private
103 	 * @param {Number} touchPoint The value where the touch is occuring on the
104 	 * strip
105 	 */
106 	SoftPot.prototype.setMinFlickMovement = function(num) {
107 		this._minFlickMovement = num;	
108 	};
109 	
110 	/**
111 	 * @private
112 	 */
113 	SoftPot.prototype.startTouch = function(touchPoint) {
114 		
115 		this._pressTimer.reset();
116 		this._pressTimer.start();
117 		
118 		// where we pressed
119 		this._touchPoint = touchPoint;
120 		this.dispatch(SoftPotEvent.PRESS);
121 		
122 		this._isTouched = true;
123 		this._isDrag = false;	
124 	};
125 
126 	/**
127 	 * @private
128 	 */
129 	SoftPot.prototype.onRelease = function() {		
130 
131 			var dispatchedFlick = false; 
132 			
133 			// discard unintentional touch / noise
134 			if (this._pressTimer.currentCount > DEBOUNCE_TIMEOUT / PRESS_TIMER_INTERVAL) {
135 				// must meet minimum time requirement for flick  
136 				if (this._flickTimer.running) {
137 					if (this._flickDir > 0) {
138 						this.dispatch(SoftPotEvent.FLICK_DOWN);
139 					} else {
140 						this.dispatch(SoftPotEvent.FLICK_UP);
141 					}	
142 					dispatchedFlick = true; 
143 					
144 				} 
145 							
146 				if (!dispatchedFlick) {
147 					// Check for presses  
148 					if (this._pressTimer.running) {
149 				
150 						// If less than tap timeout, then it is a tap 
151 						if (!this._isDrag && this._pressTimer.currentCount <= this._tapTimeout / PRESS_TIMER_INTERVAL) {
152 							this.dispatch(SoftPotEvent.TAP);
153 						}
154 					} 
155 				}
156 			}
157 
158 			this.dispatch(SoftPotEvent.RELEASE);
159 
160 			this.resetForNext(); 
161 	};
162 	
163 	/**
164 	 * @private
165 	 * @param {Number} touchPoint The value where the touch is occuring on the
166 	 * strip
167 	 */
168 	SoftPot.prototype.onMove = function(touchPoint) {		
169 	
170 		this._touchPoint = touchPoint;		
171 		// Save current point
172 		var curMovePoint = touchPoint;
173 		
174 		// Flick handeling 
175 		this._flickDistance = Math.abs(curMovePoint - this._lastMovePoint) 
176 		
177 		if (!this._isDrag && this._flickDistance > this._minFlickMovement ) {
178 			this._flickTimer.reset(); 
179 			this._flickTimer.start(); 
180 			
181 			if(curMovePoint - this._lastMovePoint > 0) {
182 				this._flickDir = 1;
183 			} else {
184 				this._flickDir = -1;
185 			}
186 			
187 			this._isDrag = false; 
188 		}			
189 		
190 		var dragDistance = Math.abs(curMovePoint - this._lastMovePoint);				
191 
192 		// Dragging handler 
193 		// Don't check when flick timer is running
194 		//console.log("min drag = " + this._minDragMovement);
195 		if ((dragDistance > this._minDragMovement) && (this._flickTimer.running === false)) {
196 			this._isDrag = true; 
197 		}
198 		
199 		if (this._isDrag) {
200 			this.dispatch(SoftPotEvent.DRAG);
201 			this._distanceFromPressed = curMovePoint - this._lastMovePoint;
202 		}
203 								
204 		this.debug("SoftPot: distance traveled flick is " + this._flickDistance); 
205 		this.debug("SoftPot: distance traveled drag is " + dragDistance); 
206 
207 		// Reuse for next 
208 		this._lastMovePoint = curMovePoint; 
209 	};
210 
211 	/**
212 	 * Scale from the minimum and maximum input values to 0.0 -> 1.0.
213 	 *
214 	 * @param {Number} minimum The minimum value
215 	 * @param {Number} maximum The maximum value
216 	 */
217 	SoftPot.prototype.setRange = function(minimum, maximum) {
218 		this._pin.removeAllFilters();
219 		this._pin.addFilter(new Scaler(mimimum, maximum, 0, 1, Scaler.LINEAR));	
220 	};
221 
222 	/**
223 	 * @private
224 	 * @type {Event} type The event type
225 	 */
226 	SoftPot.prototype.dispatch = function(type) {
227 		this.debug("SoftPot dispatch " + type);
228 		this.dispatchEvent(new SoftPotEvent(type, this._touchPoint));	
229 	};
230 
231 	/**
232 	 * Reset whenever you need the next Touch point.
233 	 * @private
234 	 */
235 	SoftPot.prototype.resetForNext = function() {
236 		this._flickTimer.stop();
237 		this._pressTimer.stop();
238 		this._isTouched = false;
239 		this._isDrag = false;
240 	};

250 		}
251 	};
