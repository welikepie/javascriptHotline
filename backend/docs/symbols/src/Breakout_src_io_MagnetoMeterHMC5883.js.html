 29 	 * @exports MagnetometerHMC5883 as BO.io.MagnetometerHMC5883

 34 	 * @constructor
 35 	 * @augments BO.I2CBase
 36 	 * @param {IOBoard} board The IOBoard instance
 37 	 * @param {Number} address The i2c address of the compass module
 38 	 * @param {Number} numSamples The number of samples averaged per 
 39 	 * measurement output. Options are: MagnetometerHMC5883.SAMPLES_1,

 45 	MagnetometerHMC5883 = function(board, address, numSamples, outputRate) {
 46 		address = address || MagnetometerHMC5883.DEVICE_ID;
 47 		numSamples = numSamples || MagnetometerHMC5883.SAMPLES_1;
 48 		outputRate = outputRate || MagnetometerHMC5883.HZ_30;
 49 
 50 		I2CBase.call(this, board, address);
 51 
 52 		this._x = 0;
 53 		this._y = 0;
 54 		this._z = 0;
 55 
 56 		// To do: scale is currently fixed. allow user to change this value?
 57 		this._scale = 0.92; // mG/LSb
 58 
 59 		this._isReading = false;
 60 		this._debugMode = BO.enableDebugging;
 61 
 62 		this.name = "MagnetometerHMC5883";
 63 		
 64 		var measurement = 0x00;
 65 		var CRAVal = (numSamples << 5) | (outputRate << 2)  | measurement;
 66 			
 67 		// 1 sample, continuous measurement rate 30 Hz, normal measurement config
 68 		this.sendI2CRequest([I2CBase.WRITE, this.address, CRA, CRAVal]);
 69 		// startup in continuous measurement mode
 70 		this.sendI2CRequest([I2CBase.WRITE, this.address, MODE, 0x00]);
 71 		
 72 		this.startReading();
 73 
 74 	};

113 	/**
114 	 * @private
115 	 */
116 	MagnetometerHMC5883.prototype.handleI2C = function(data) {
117 		var xVal,
118 			yVal,
119 			zVal;
120 
121 		// data[0] = register
122 		if (data[0] === DATAX0) {
123 			xVal = (data[1] << 8) | data[2];
124 			zVal = (data[3] << 8) | data[4];
125 			yVal = (data[5] << 8) | data[6];
126 
127 			// correct for negative number
128 			if(xVal >> 15) {
129 				this._x = ((xVal ^ 0xFFFF) + 1) * -1;
130 			} else this._x = xVal;
131 			if(yVal >> 15) {
132 				this._y = ((yVal ^ 0xFFFF) + 1) * -1;
133 			} else this._y = yVal;
134 			if(zVal >> 15) {
135 				this._z = ((zVal ^ 0xFFFF) + 1) * -1;
136 			} else this._z = zVal;
137 
138 			// a value of -4096 indicates an ADC overflow or underflow
139 		
140 			this.dispatchEvent(new MagnetometerEvent(MagnetometerEvent.UPDATE));
141 		} else {
142 			console.log("Warning: MagnetometerHMC5883 received data from unknown register");
143 		}
144 	};

146 	/**
147 	 * @private
148 	 */
149 	MagnetometerHMC5883.prototype.getHeading = function(x, y) {
150 		var heading = 0.0;
151 
152 		// algorithm from Applications of Magnetoresistive Sensors in Navigation Systems
153 		// by Michael J. Caruso of Honeywell Inc.
154 		if (y > 0) heading = 90.0 - Math.atan(x/y) * 180/Math.PI;
155 		else if (y < 0) heading = 270.0 - Math.atan(x/y) * 180/Math.PI;
156 		else if (y === 0 && x < 0) heading = 180.0;
157 		else if (y === 0 && x > 0) heading = 0.0;
158 
159 		// alternate algorithm
160 		// heading = Math.atan2(y, x);
161 		// if (heading < 0) heading += 2*Math.PI;
162 		// if (heading > 2*Math.PI) heading -= 2*Math.PI;
163 		// return heading * RAD_TO_DEG;	
164 
165 		return heading;
166 	};
