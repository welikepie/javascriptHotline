 12 JSUTILS.namespace('BO.io.LED');
 13 
 14 BO.io.LED = (function () {
 15 
 16     var LED;
 17 
 18     // Dependencies
 19     var Pin = BO.Pin;
 20     var Oscillator = BO.generators.Oscillator;
 21 
 22     /**
 23      * <p>PLEASE NOTE: To use the fade methods, or to use an waveform other 
 24      * than Oscillator.SQUARE the LED must be connected to a PWM pin.</p>
 25      *
 26      * <p>SOURCE_DRIVE vs SYNC_DRIVE. If the Anode (longer LED pin) is
 27      * connected to the microcontroller pin, then it is SOURCE_DRIVE. If the
 28      * Cathode is connected to the microcontroller pin, then it is 
 29      * SYNC_DRIVE.</p>
 30      *
 31      * @exports LED as BO.io.LED
 32      * @class Creates an interface to an LED. This object provides helpful
 33      * methods for blinking and fading LEDs. To use the fading methods, the
 34      * LED must be connected to a PWM pin on the I/O board.
 35      * @param {IOBoard} board A reference to the IOBoard the LED is attached to.
 36      * @param {Pin} ledPin A reference to the Pin the LED is connected to.
 37      * @param {Number} driveMode The drive mode of the LED. Must be set to
 38      * LED.SOURCE_MODE or LED.SYNC_MODE. SOURCE_MODE is default.
 39      */
 40     LED = function (board, ledPin, driveMode) {
 41         "use strict";
 42         
 43         this.name = "LED";
 44 
 45         this._driveMode = driveMode || LED.SOURCE_DRIVE;
 46         this._pin = ledPin;
 47         this._onValue = 1;
 48         this._offValue = 0;
 49         this._supportsPWM = undefined;
 50 
 51         if (this._driveMode === LED.SOURCE_DRIVE) {
 52             this._onValue = 1;
 53             this._offValue = 0;
 54         } else if (this._driveMode === LED.SYNC_DRIVE) {
 55             this._onValue = 0;
 56             this._offValue = 1;
 57         } else {
 58             throw "driveMode should be LED.SOURCE_DRIVE or LED.SYNC_DRIVE";
 59         }
 60 
 61         // If the pin supports PWM, set PWM mode else set DOUT mode
 62         if (this._pin.getCapabilities()[Pin.PWM]) {
 63             board.setDigitalPinMode(this._pin.number, Pin.PWM);
 64             this._supportsPWM = true;
 65         } else {
 66             board.setDigitalPinMode(this._pin.number, Pin.DOUT);
 67             this._supportsPWM = false;
 68         }
 69 
 70         // Start in the off state
 71         this.off();
 72     };
 73 
 74     LED.prototype = {
 75 
 76         /**
 77          * Get or set the current value (intensity) of the LED.
 78          * 
 79          * @name LED#intensity
 80          * @property
 81          * @type Number
 82          */ 
 83         get intensity() {
 84             return this._pin.value;
 85         },
 86         set intensity(val) {
 87             // If the pin does not support PWM, force the value to 1 or 0
 88             if (!this._supportsPWM) {
 89                 if (val < 0.5) {
 90                     val = 0;
 91                 } else {
 92                     val = 1;
 93                 }
 94             }
 95 
 96             if (this._driveMode === LED.SOURCE_DRIVE) {
 97                 this._pin.value = val;
 98             } else if (this._driveMode === LED.SYNC_DRIVE) {
 99                 this._pin.value = 1 - val;
100             }
101         },
102         
103         /**
104          * Turn the LED on.
105          */
106         on: function () {
107             this._pin.value = this._onValue;
108         },
109 
110         /**
111          * Turn the LED off.
112          */
113         off: function () {
114             this._pin.value = this._offValue;
115         },
116 
117         /**
118          * Check if the LED is on.
119          * 
120          * @return {Boolean} True if the LED is on, false if it is off.
121          */
122         isOn: function () {
123             return this._pin.value === this._onValue;
124         },
125 
126         /**
127          * Toggle the LED on or off
128          */
129         toggle: function () {
130             this._pin.value = 1 - this._pin.value;
131         },
132 
133         /**
134          * @param {Number} interval The time interval to blink the LED.
135          * @param {Number} times The number of times the LED should blink.
136          * A value of 0 will blink forever.
137          * @param {Function} wave The waveform to apply (default is Oscillator.SQUARE)
138          * @see BO.generator.Oscillator
139          */
140         blink: function (interval, times, wave) {
141             var freq = 1000 / interval;
142             times = times || 0;
143             wave = wave || Oscillator.SQUARE;
144 
145             if (!this._supportsPWM && wave !== Oscillator.SQUARE) {
146                 console.log("warning: Only Oscillator.SQUARE may be used on a non-PWM pin.");
147                 console.log("debug: Setting wave to Oscillator.SQUARE.");
148                 wave = Oscillator.SQUARE;
149             }
150 
151             //var osc = new Oscillator(wave, freq, 1, 0, 0, times);
152             this._pin.addGenerator(new Oscillator(wave, freq, 1, 0, 0, times));
153             //osc.start();
154             this._pin.generator.start();
155         },
156 
157         /**
158          * Stop the LED blink cycle.
159          */
160         stopBlinking: function () {
161             if (this._pin.generator !== null) {
162                 this._pin.generator.stop();
163             }
164             this.off();
165         },
166 
167         /**
168          * The LED must be connected to a PWM pin to use this method.
169          *
170          * @param {Number} time The fade-in time (in milliseconds).
171          */
172         fadeIn: function (time) {
173             this.fadeTo(this._onValue, time);
174         },
175 
176         /**
177          * The LED must be connected to a PWM pin to use this method.
178          *
179          * @param {Number} time The fade-out time (in milliseconds).
180          */
181         fadeOut: function (time) {
182             this.fadeTo(this._offValue, time);
183         },
184 
185         /**
186          * The LED must be connected to a PWM pin to use this method.
187          *      
188          * @param {Number} to The new intensity value to fade to.
189          * @param {Number} time The fade time (in milliseconds).
190          */
191         fadeTo: function (to, time) {
192 
193             if (!this._supportsPWM) {
194                 console.log("warning: Fade methods can only be used for LEDs connected to PWM pins.");
195                 return;
196             }
197 
198             if (this._driveMode === LED.SYNC_DRIVE) {
199                 to = 1 - to;
200             }
201 
202             time = time || 1000;
203             var freq = 1000 / time;
204             if (this._pin.value !== to) {
205                 this._pin.addGenerator(new Oscillator(Oscillator.LINEAR, freq, to - this._pin.value, this._pin.value, 0, 1));
206                 this._pin.generator.start();
207             } else {
208                 this._pin.removeGenerator();
209             }       
210         }   
211     };
212 
213     LED.SOURCE_DRIVE = 0;
214     LED.SYNC_DRIVE = 1;
215 
216     return LED;
217 }());
