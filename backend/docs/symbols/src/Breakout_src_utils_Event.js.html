  6 JSUTILS.namespace('JSUTILS.Event');
  7 
  8 JSUTILS.Event = (function() {
  9 
 10 	var Event;
 11 
 12 	/** 
 13 	 * A base class for the creation of Event objects.
 14 	 *
 15 	 * @class A base class for the creation of Event objects.
 16 	 * @exports Event as JSUTILS.Event
 17 	 * @param {String} type event type
 18 	 */
 19 	Event = function(type) {
 20 
 21 		this._type = type;
 22 		this._target = null;
 23 
 24 		this.name = "Event";
 25 	};
 26 
 27 	Event.prototype = {
 28 		/**
 29 		 * The event type
 30 		 * @name Event#type
 31 		 * @type String
 32 		 */ 		
 33 		get type() {
 34 			return this._type;
 35 		},
 36 		set type(val) {
 37 			this._type = val;
 38 		},
 39 
 40 		/**
 41 		 * The event target
 42 		 * @name Event#target
 43 		 * @type Object
 44 		 */ 
 45 		get target() {
 46 			return this._target;
 47 		},
 48 		set target(val) {
 49 			this._target = val;
 50 		}
 51 
 52 	};
 53 
 54 	// Generic events
 55 
 56 	/** 
 57 	 * Description
 58 	 * @constant
 59 	 */
 60 	Event.CONNECTED = "connected";
 61 	/** 
 62 	 * Description
 63 	 * @constant
 64 	 */
 65 	Event.CHANGE	= "change";
 66 	/** 
 67 	 * Description
 68 	 * @constant
 69 	 */
 70 	Event.COMPLETE	= "complete";
 71 
 72 	return Event;
 73 
 74 }());
