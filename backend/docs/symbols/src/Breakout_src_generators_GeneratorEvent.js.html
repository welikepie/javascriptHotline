 12  JSUTILS.namespace('BO.generators.GeneratorEvent');
 13 
 14  BO.generators.GeneratorEvent = (function() {
 15  	"use strict";
 16 
 17  	var GeneratorEvent;
 18 
 19  	// dependencies
 20 	var Event = JSUTILS.Event;
 21 
 22  	/**
 23  	 * @exports GeneratorEvent as BO.generators.GeneratorEvent
 24 	 * @class An Event object to be dispatched (fired) by a Generator
 25 	 * object when its value has updated.	 	 
 26  	 * @constructor
 27  	 * @augments JSUTILS.Event
 28  	 * @param {String} type The event type
 29  	 */
 30  	GeneratorEvent = function(type) {
 31  		
 32  		Event.call(this, type);
 33 
 34  		this.name = "GeneratorEvent";
 35  	};
 36 
 37  	GeneratorEvent.prototype = JSUTILS.inherit(Event.prototype);
 38  	GeneratorEvent.prototype.constructor = GeneratorEvent;
 39 
 40  	/** @constant */
 41  	GeneratorEvent.UPDATE = "update";
 42 
 43  	return GeneratorEvent;
 44 
 45  }());
