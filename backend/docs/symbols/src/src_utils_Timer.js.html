  6 JSUTILS.namespace('JSUTILS.Timer');
  7 
  8 JSUTILS.Timer = (function () {
  9 
 10     var Timer;
 11 
 12     // Dependencies
 13     var TimerEvent = JSUTILS.TimerEvent,
 14         EventDispatcher = JSUTILS.EventDispatcher;
 15 
 16     /**
 17      * An as3-like Timer object.
 18      *
 19      * @class The Timer object wraps the window.setInterval() method to provide
 20      * an as3-like Timer interface.
 21      * @exports Timer as JSUTILS.Timer
 22      * @extends JSUTILS.EventDispatcher  
 23      * @param {Number} delay The delay (ms) interval between ticks
 24      * @param {Number} repeatCount The number of number of ticks.
 25      * A value of zero will set the timer to repeat forever. Default = 0
 26      */
 27     Timer = function (delay, repeatCount) {
 28 
 29         EventDispatcher.call(this, this);
 30 
 31         this.name = "Timer";
 32 
 33         this._count = 0;
 34         this._delay = delay;
 35         this._repeatCount = repeatCount || 0;
 36         this._isRunning = false;
 37 
 38         this._timer = null;
 39     };
 40 
 41     Timer.prototype = JSUTILS.inherit(EventDispatcher.prototype);
 42     Timer.prototype.constructor = Timer;
 43 
 44 
 45     /**
 46      * The delay interval in milliseconds.
 47      * 
 48      * @name Timer#delay
 49      * @type Number
 50      */ 
 51     Timer.prototype.__defineGetter__("delay", function () {
 52         return this._delay;
 53     });
 54     Timer.prototype.__defineSetter__("delay", function (val) { 
 55         this._delay = val;
 56         if (this._isRunning) {
 57             this.stop();
 58             this.start();
 59         }
 60     }); 
 61 
 62     /**
 63      * The repeat count in milliseconds.
 64      * 
 65      * @name Timer#repeatCount
 66      * @type Number
 67      */ 
 68     Timer.prototype.__defineGetter__("repeatCount", function () {
 69         return this._repeatCount;
 70     });
 71     Timer.prototype.__defineSetter__("repeatCount", function (val) { 
 72         this._repeatCount = val;
 73         if (this._isRunning) {
 74             this.stop();
 75             this.start();
 76         }
 77     });
 78 
 79     /**
 80      * [read-only] Returns true if the timer is running.
 81      * 
 82      * @name Timer#running
 83      * @type Number
 84      */ 
 85     Timer.prototype.__defineGetter__("running", function () {
 86         return this._isRunning;
 87     });
 88 
 89     /**
 90      * [read-only] Returns the current count (number of ticks since timer
 91      * started).
 92      * 
 93      * @name Timer#currentCount
 94      * @type Number
 95      */ 
 96     Timer.prototype.__defineGetter__("currentCount", function () {
 97         return this._count;
 98     });
 99 
100     /**
101      * Start the timer.
102      */
103     Timer.prototype.start = function () {
104         if (this._timer === null) {
105             this._timer = setInterval(this.onTick.bind(this), this._delay);
106             this._isRunning = true;
107         }
108     };
109 
110     /**
111      * Stop the timer and reset the count to zero.
112      */
113     Timer.prototype.reset = function () {
114         this.stop();
115         this._count = 0;
116     };
117 
118     /**
119      * Stop the timer.
120      */
121     Timer.prototype.stop = function () {
122         if (this._timer !== null) {
123             clearInterval(this._timer);
124             this._timer = null;
125             this._isRunning = false;
126         }
127     };
128 
129     /**
130      * Description
131      * @private
132      */
133     Timer.prototype.onTick = function () {
134         this._count = this._count + 1;
135         if (this._repeatCount !== 0 && this._count > this._repeatCount) {
136             this.stop();
137             this.dispatchEvent(new TimerEvent(TimerEvent.TIMER_COMPLETE));
138         } else {
139             this.dispatchEvent(new TimerEvent(TimerEvent.TIMER));
140         }
141     };
142 
143     // Document events
144 
145     /**
146      * The timerTick event is dispatched at the rate specified 
147      * by the delay interval.
148      * @name Timer#timerTick
149      * @type JSUTILS.TimerEvent.TIMER
150      * @event
151      * @param {JSUTILS.Timer} target A reference to the Timer object.
152      */ 
153 
154     /**
155      * The timerComplete event is dispatched when the repeatCount value
156      * has been reached.
157      * @name Timer#timerComplete
158      * @type JSUTILS.TimerEvent.TIMER_COMPLETE
159      * @event
160      * @param {JSUTILS.Timer} target A reference to the Timer object.
161      */      
162 
163     return Timer;
164 
165 }());
