 14 BO.Pin = (function() {
 15 	//"use strict";	// closure compiler doesn't like this... should be fixed soon though

 19 	// dependencies
 20 	var EventDispatcher = JSUTILS.EventDispatcher,
 21 		PinEvent = BO.PinEvent,

 24 	/**
 25 	 * An object to represent an IOBoard pin
 26 	 *
 27 	 * @exports Pin as BO.Pin
 28 	 * @class Each analog and digital pin of the physical I/O board is represented by a Pin object.
 29 	 * The Pin object is the foundation for many of the io objects and is also very
 30 	 * useful on its own. See the Using The Pin Object Guide on http://breakoutjs.com
 31 	 * for a detailed overview.
 32 	 * @param {Number} number The pin number
 33 	 * @param {Number} type The type of pin
 34 	 */
 35 	Pin = function(number, type) {
 36 
 37 		this.name = "Pin";
 38 
 39 		this._type = type;
 40 		this._capabilities;
 41 		this._number = number;
 42 		this._analogNumber = undefined;
 43 		this._maxPWMValue = 255;
 44 		this._value = 0;
 45 		this._lastValue = -1;
 46 		this._preFilterValue;
 47 		this._average = 0;
 48 		this._minimum = Math.pow(2, 16);
 49 		this._maximum = 0;
 50 		this._sum = 0;
 51 		this._numSamples = 0;
 52 		this._filters = null;
 53 		this._generator = null;
 54 
 55 		this._autoSetValueCallback = this.autoSetValue.bind(this);
 56 		
 57 		this._evtDispatcher = new EventDispatcher(this);	
 58 
 59 	};
 60 
 61 	Pin.prototype = {
 62 		
 63 		/**
 64 		 * The analogNumber sould only be set internally.
 65 		 * @private
 66 		 */		
 67 		setAnalogNumber: function(num) {
 68 			this._analogNumber = num;
 69 		},
 70 
 71 		/**
 72 		 * [read-only] The analog pin number used by the IOBoard (printed on board or datasheet).
 73 		 * @name Pin#analogNumber
 74 		 * @property
 75 		 * @type Number
 76 		 */	
 77 		get analogNumber() {
 78 			return this._analogNumber;
 79 		},		
 80 
 81 		/**
 82 		 * [read-only] The pin number corresponding to the Arduino documentation for the type of board.
 83 		 * @name Pin#number
 84 		 * @property
 85 		 * @type Number
 86 		 */			 
 87 		get number() {
 88 			return this._number;
 89 		},
 90 
 91 		/**
 92 		 * The maximum PWM value supported for this pin. This value should
 93 		 * normally be set internally.
 94 		 * @private
 95 		 */
 96 		setMaxPWMValue: function(val) {
 97 			this._maxPWMValue = value;
 98 		},
 99 
100 		/**
101 		 * [read-only] The maximum PWM value supported for this pin.
102 		 * <p> This is the max PWM value supported by Arduino (currently 255) rather 
103 		 * than the max PWM value specified by the microcontroller datasheet.</p>
104 		 *
105 		 * @name Pin#maxPWMValue
106 		 * @property
107 		 * @type Number
108 		 */			 
109 		get maxPWMValue() {
110 			return this._maxPWMValue;
111 		},		
112 		
113 		/**
114 		 * [read-only] The average value of the pin over time. Call clear() to reset.
115 		 * @name Pin#average
116 		 * @property
117 		 * @type Number
118 		 */			 
119 		get average() {
120 			return this._average;
121 		},
122 
123 		/**
124 		 * [read-only] The minimum value of the pin over time. Call clear() to reset.
125 		 * @name Pin#minimum
126 		 * @property
127 		 * @type Number
128 		 */			 	
129 		get minimum() {
130 			return this._minimum;
131 		},
132 		
133 		/**
134 		 * [read-only] The maximum value of the pin over time. Call clear() to reset.
135 		 * @name Pin#maximum
136 		 * @property
137 		 * @type Number
138 		 */			 
139 		get maximum() {
140 			return this._maximum;
141 		},
142 		
143 		/**
144 		 * The current digital or analog value of the pin.
145 		 * @name Pin#value
146 		 * @property
147 		 * @type Number
148 		 */		 
149 		get value() {
150 			return this._value;
151 		},
152 		set value(val) {
153 			this._lastValue = this._value;
154 			this._preFilterValue = val;
155 			this._value = this.applyFilters(val);
156 			this.calculateMinMaxAndMean(this._value);
157 			this.detectChange(this._lastValue, this._value);
158 		},
159 		
160 		/**
161 		 * [read-only] The last pin value.
162 		 * @name Pin#lastValue
163 		 * @property
164 		 * @type Number
165 		 */			 
166 		get lastValue() {
167 			return this._lastValue;
168 		},
169 		
170 		/**
171 		 * [read-only] The value before any filters were applied.
172 		 * @name Pin#preFilterValue
173 		 * @property
174 		 * @type Number
175 		 */			 
176 		get preFilterValue() {
177 			return this._preFilterValue;
178 		},
179 
180 		/**
181 		 * Get and set filters for the Pin.
182 		 * @name Pin#filters
183 		 * @property
184 		 * @type FilterBase
185 		 */	
186 		get filters() {
187 			return this._filters;
188 		},
189 		set filters(filterArray) {
190 			this._filters = filterArray;
191 		},
192 
193 		/**
194 		 * [read-only] Get a reference to the current generator.
195 		 * @name Pin#generator
196 		 * @property
197 		 * @type GeneratorBase
198 		 */	
199 		get generator() {
200 			return this._generator;
201 		},
202 
203 		/**
204 		 * The type/mode of the pin (0: DIN, 1: DOUT, 2: AIN, 3: AOUT, 
205 		 * 4: PWM, 5: SERVO, 6: SHIFT, 7: I2C). Use 
206 		 * IOBoard.setDigitalPinMode(pinNumber) to set the pin type.
207 		 * @return {Number} The pin type/mode
208 		 */	
209 		getType: function() {
210 			return this._type;
211 		},
212 
213 		/**
214 		 * Set the pin type. This method should only be used internally.
215 		 * @private
216 		 */
217 		setType: function(pinType) {
218 			// Ensure pin type is valid
219 			if (pinType >= 0 && pinType < Pin.TOTAL_PIN_MODES) {
220 				this._type = pinType;
221 			}
222 		},			
223 
224 		/**
225 		 * An object storing the capabilities of the pin.
226 		 * @return {Object} An object describing the capabilities of this Pin.
227 		 */	
228 		getCapabilities: function() {
229 			return this._capabilities;
230 		},
231 
232 		/**
233 		 * This method should only be used internally.
234 		 * @private
235 		 */
236 		setCapabilities: function(pinCapabilities) {
237 			this._capabilities = pinCapabilities;
238 		},		
239 
240 		/**
241 		 * Dispatch a Change event whenever a pin value changes
242 		 * @private
243 		 */
244 		detectChange: function(oldValue, newValue) {
245 			if (oldValue === newValue) return;
246 			this.dispatchEvent(new PinEvent(PinEvent.CHANGE));
247 
248 			if (oldValue <= 0 && newValue !== 0) {
249 				this.dispatchEvent(new PinEvent(PinEvent.RISING_EDGE));
250 			} else if (oldValue !== 0 && newValue <=0) {
251 				this.dispatchEvent(new PinEvent(PinEvent.FALLING_EDGE));
252 			}
253 		},
254 		
255 		/**
256 		 * From funnel Pin.as
257 		 * @private
258 		 */
259 		clearWeight: function() {
260 			this._sum = this._average;
261 			this._numSamples = 1;
262 		},
263 		
264 		/**
265 		 * From funnel Pin.as
266 		 * @private
267 		 */
268 		calculateMinMaxAndMean: function(val) {
269 			var MAX_SAMPLES = Number.MAX_VALUE;
270 
271 			this._minimum = Math.min(val, this._minimum);
272 			this._maximum = Math.max(val, this._maximum);
273 			
274 			this._sum += val;
275 			this._average = this._sum / (++this._numSamples);
276 			if (this._numSamples >= MAX_SAMPLES) {
277 				this.clearWeight();
278 			}
279 		},
280 			
281 		/**
282 		 * Resets the minimum, maximum, average and lastValue of the pin.
283 		 */
284 		clear: function() {
285 			this._minimum = this._maximum = this._average = this._lastValue = this._preFilterValue;
286 			this.clearWeight();
287 		},
288 		
289 		/**
290 		 * Add a new filter to the Pin.
291 		 * @param {FilterBase} newFilter A filter object that extends FilterBase.
292 		 * @see BO.filters.Convolution
293 		 * @see BO.filters.Scaler
294 		 * @see BO.filters.TriggerPoint
295 		 */
296 		addFilter: function(newFilter) {
297 
298 			if (newFilter === null) {
299 				return;
300 			}
301 
302 			if (this._filters === null) {
303 				this._filters = [];
304 			}
305 
306 			this._filters.push(newFilter);
307 		},
308 
309 		/**
310 		 * Add a new generator to the Pin. A pin can only have one generator
311 		 * assigned. 
312 		 * Assigning a new generator will replace the previously assigned 
313 		 * generator.
314 		 *
315 		 * @param {GeneratorBase} newGenerator A generator object that extends GeneratorBase.
316 		 * @see BO.generators.Oscillator
317 		 */
318 		addGenerator: function(newGenerator) {
319 			this.removeGenerator();
320 			this._generator = newGenerator;
321 			this._generator.addEventListener(GeneratorEvent.UPDATE, this._autoSetValueCallback);
322 		},
323 
324 		/**
325 		 * Removes the generator from the pin.
326 		 */
327 		removeGenerator: function() {
328 			if (this._generator !== null) {
329 				this._generator.removeEventListener(GeneratorEvent.UPDATE, this._autoSetValueCallback);
330 			}
331 			this._generator = null;				
332 		},
333 
334 		/**
335 		 * Removes all filters from the pin.
336 		 */
337 		removeAllFilters: function() {
338 			this._filters = null;
339 		},
340 
341 		/**
342 		 * @private
343 		 */
344 		autoSetValue: function(event) {
345 			var val = this._generator.value;
346 			this.value = val;
347 		},
348 
349 		/**
350 		 * @private
351 		 */
352 		applyFilters: function(val) {
353 			var result;
354 
355 			if (this._filters === null) return val;
356 			
357 			result = val;
358 			var len = this._filters.length;
359 			for (var i = 0; i < len; i++) {
360 				result = this._filters[i].processSample(result);
361 			}
362 
363 			return result;
364 		},
365 
366 		// Implement EventDispatcher
367 		
368 		/**
369 		 * @param {String} type The event type
370 		 * @param {Function} listener The function to be called when the event is fired
371 		 */
372 		addEventListener: function(type, listener) {
373 			this._evtDispatcher.addEventListener(type, listener);
374 		},
375 		
376 		/**
377 		 * @param {String} type The event type
378 		 * @param {Function} listener The function to be called when the event is fired
379 		 */
380 		removeEventListener: function(type, listener) {
381 			this._evtDispatcher.removeEventListener(type, listener);
382 		},
383 		
384 		/**
385 		 * @param {String} type The event type
386 		 * return {boolean} True is listener exists for this type, false if not.
387 		 */
388 		hasEventListener: function(type) {
389 			return this._evtDispatcher.hasEventListener(type);
390 		},
391 
392 		/**
393 		 * @param {PinEvent} type The Event object
394 		 * @param {Object} optionalParams Optional parameters to assign to the event object.
395 		 * return {boolean} True if dispatch is successful, false if not.
396 		 */	
397 		dispatchEvent: function(event, optionalParams) {
398 			return this._evtDispatcher.dispatchEvent(event, optionalParams);
399 		}		
400
