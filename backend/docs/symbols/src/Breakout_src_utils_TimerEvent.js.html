  6 JSUTILS.namespace('JSUTILS.TimerEvent');
  7 
  8 JSUTILS.TimerEvent = (function() {
  9 
 10 	var TimerEvent;
 11 
 12 	// Dependencies
 13 	var Event = JSUTILS.Event;
 14 
 15 	/**
 16 	 * TimerEvent Description.
 17 	 * 
 18 	 * @class An Event object to be dispatched (fired) by a Timer object.	
 19 	 * @exports TimerEvent as JSUTILS.TimerEvent	 
 20 	 * @extends JSUTILS.Event
 21  	 * @param {String} type The event type	 
 22 	 */
 23 	TimerEvent = function(type) {
 24 
 25 		this.name = "TimerEvent";
 26 
 27 		Event.call(this, type);
 28 	};
 29 
 30 	/** 
 31 	 * Description
 32 	 * @constant
 33 	 */
 34 	TimerEvent.TIMER = "timerTick";
 35 	/** 
 36 	 * Description
 37 	 * @constant
 38 	 */
 39 	TimerEvent.TIMER_COMPLETE = "timerComplete";
 40 
 41 	TimerEvent.prototype = JSUTILS.inherit(Event.prototype);
 42 	TimerEvent.prototype.constructor = TimerEvent;
 43 
 44 	return TimerEvent;
 45 
 46 }());
