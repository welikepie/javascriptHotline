 57     /**

 62      * @exports IOBoard as BO.IOBoard

185         /**
186          * @private
187          */
188         function processInput(inputData) {
189             inputData *= 1; // Force inputData to integer (is there a better way to do this?)
190             var len;
191 
192             _inputDataBuffer.push(inputData);
193             len = _inputDataBuffer.length;
194 
195             if (_inputDataBuffer[0] >= 128 && _inputDataBuffer[0] != START_SYSEX) {
196                 if (len === 3) {
197                     processMultiByteCommand(_inputDataBuffer);
198                     // Clear buffer
199                     _inputDataBuffer = [];
200                 }
201             } else if (_inputDataBuffer[0] === START_SYSEX && _inputDataBuffer[len - 1] === END_SYSEX) {
202                 processSysexCommand(_inputDataBuffer);
203                 // Clear buffer
204                 _inputDataBuffer = [];
205             } else if (inputData >= 128 && _inputDataBuffer[0] < 128) {
206                 // If for some reason we got a new command and there is already data
207                 // in the buffer, reset the buffer
208                 console.log("warning: Malformed input data... resetting buffer");
209                 _inputDataBuffer = [];
210                 if (inputData !== END_SYSEX) {
211                     _inputDataBuffer.push(inputData);
212                 }
213             }
214         }

216         /**
217          * @private
218          */
219         function processMultiByteCommand(commandData) {
220             var command = commandData[0],
221                 channel;
222 
223             if (command < 0xF0) {
224                 command = command & 0xF0;
225                 channel = commandData[0] & 0x0F;
226             }
227 
228             switch (command) {
229             case DIGITAL_MESSAGE:
230                 processDigitalMessage(channel, commandData[1], commandData[2]); //(LSB, MSB)
231                 break;
232             case REPORT_VERSION:
233                 _firmwareVersion = commandData[1] + commandData[2] / 10;
234                 _self.dispatchEvent(new IOBoardEvent(IOBoardEvent.FIRMWARE_VERSION), {version: _firmwareVersion});
235                 break;
236             case ANALOG_MESSAGE:
237                 processAnalogMessage(channel, commandData[1], commandData[2]);
238                 break;
239             }
240         }

242         /**
243          * @private
244          */
245         function processDigitalMessage(port, bits0_6, bits7_13) {
246             var offset = port * 8,
247                 lastPin = offset + 8,
248                 portVal = bits0_6 | (bits7_13 << 7),
249                 pinVal,
250                 pin = {};
251             
252             if (lastPin >= _totalPins) {
253                 lastPin = _totalPins;
254             }
255             
256             var j = 0;
257             for (var i = offset; i < lastPin; i++) {
258                 pin = _self.getDigitalPin(i);
259                 // Ignore data send on Firmata startup
260                 if (pin === undefined) {
261                     return;
262                 }
263                 
264                 if (pin.getType() == Pin.DIN) {
265                     pinVal = (portVal >> j) & 0x01;
266                     if (pinVal != pin.value) {
267                         pin.value = pinVal;
268                         _self.dispatchEvent(new IOBoardEvent(IOBoardEvent.DIGITAL_DATA), {pin: pin});
269                     }
270                 }
271                 j++;
272             }
273         }

275         /**
276          * @private
277          */
278         function processAnalogMessage(channel, bits0_6, bits7_13) {
279             var analogPin = _self.getAnalogPin(channel);
280 
281             // NOTE: Is there a better way to handle this? This issue is on
282             // browser refresh the IOBoard board is still sending analog data
283             // if analog reporting was set before the refresh. Analog reporting
284             // won't be disabled by systemReset systemReset() is called. There
285             // is not a way to call that method fast enough so the following
286             // code is needed. An alternative would be to set a flag that
287             // prevents critical operations before systemReset has completed.
288             if (analogPin === undefined) {
289                 return;
290             }
291             // Map analog input values from 0-1023 to 0.0-1.0
292             // To Do: Add maxADCValue property to Pin or IOBoard object to
293             // support ADC values > 1023?
294             // maxADCValue could be set during configuration routine if it's
295             // supported by Firmata in the future.
296             analogPin.value = _self.getValueFromTwo7bitBytes(bits0_6, bits7_13) / 1023;
297             if (analogPin.value != analogPin.lastValue) {
298                 _self.dispatchEvent(new IOBoardEvent(IOBoardEvent.ANALOG_DATA), {pin: analogPin});
299             }
300         }

302         /**
303          * @private
304          */
305         function processSysexCommand(sysexData) {
306             // Remove the first and last element from the array
307             // since these are the START_SYSEX and END_SYSEX 
308             sysexData.shift();
309             sysexData.pop();
310 
311             var command = sysexData[0];
312             switch (command) {
313             case REPORT_FIRMWARE:
314                 processQueryFirmwareResult(sysexData);
315                 break;
316             case STRING_DATA:
317                 processSysExString(sysexData);
318                 break;
319             case CAPABILITY_RESPONSE:
320                 processCapabilitiesResponse(sysexData);
321                 break;
322             case PIN_STATE_RESPONSE:
323                 processPinStateResponse(sysexData);
324                 break;
325             case ANALOG_MAPPING_RESPONSE:
326                 processAnalogMappingResponse(sysexData);
327                 break;
328             default:
329                 // Custom sysEx message
330                 _self.dispatchEvent(new IOBoardEvent(IOBoardEvent.SYSEX_MESSAGE), {message: sysexData});
331                 break;
332             }
333         }

335         /**
336          * @private
337          */
338         function processQueryFirmwareResult(msg) {
339             var data;
340             for (var i = 3; i < msg.length; i += 2)
341             {
342                 data = String.fromCharCode(msg[i]);
343                 data += String.fromCharCode(msg[i + 1]);
344                 _firmwareName += data;
345             }
346             _firmwareVersion = msg[1] + msg[2] / 10;
347             _self.dispatchEvent(new IOBoardEvent(IOBoardEvent.FIRMWARE_NAME), {name: _firmwareName, version: _firmwareVersion});
348         }

350         /**
351          * @private
352          */
353         function processSysExString(msg) {
354             var str = "",
355                 data,
356                 len = msg.length;
357 
358             for (var i = 1; i < len; i += 2) {
359                 data = String.fromCharCode(msg[i]);
360                 data += String.fromCharCode(msg[i + 1]);
361                 str += data.charAt(0);
362             }
363             _self.dispatchEvent(new IOBoardEvent(IOBoardEvent.STRING_MESSAGE), {message: str});
364         }

366         /** 
367          * Auto configure using capabilities response.
368          * This creates a configuration for any board in the Firmata boards.h 
369          * file.
370          *
371          * @private
372          */
373         function processCapabilitiesResponse(msg) {
374             // If running in multi-client mode and this client is already 
375             // configured, ignore capabilities response
376             if (_isConfigured) {
377                 return;
378             }
379 
380             var pinCapabilities = {},
381                 byteCounter = 1, // Skip 1st byte because it's the command
382                 pinCounter = 0,
383                 analogPinCounter = 0,
384                 len = msg.length,
385                 type,
386                 pin;
387                     
388             // Create default configuration
389             while (byteCounter <= len) {
390                 // 127 denotes end of pin's modes
391                 if (msg[byteCounter] == 127) {
392                     
393                     // Is digital pin mapping even necessary anymore?
394                     _digitalPinMapping[pinCounter] = pinCounter;
395                     type = undefined;
396                     
397                     // Assign default types
398                     if (pinCapabilities[Pin.DOUT]) {
399                         // Map digital pins
400                         type = Pin.DOUT;
401                     }
402                     
403                     if (pinCapabilities[Pin.AIN]) {
404                         type = Pin.AIN;
405                         // Map analog input pins
406                         _analogPinMapping[analogPinCounter++] = pinCounter;
407                     }
408                     
409                     pin = new Pin(pinCounter, type);
410                     pin.setCapabilities(pinCapabilities);
411                     managePinListener(pin);
412                     _ioPins[pinCounter] = pin;
413                     
414                     // Store the 2 i2c pin numbers if they exist
415                     // To Do: allow for more than 2 i2c pins on a board?
416                     // How to identify SDA-SCL pairs in that case?
417                     if (pin.getCapabilities()[Pin.I2C]) {
418                         _i2cPins.push(pin.number);
419                     }
420                     
421                     pinCapabilities = {};
422                     pinCounter++;
423                     byteCounter++;
424                 } else {
425                     // Create capabilities object (mode: resolution) for each 
426                     // mode supported by each pin
427                     pinCapabilities[msg[byteCounter]] = msg[byteCounter + 1];
428                     byteCounter += 2;
429                 }
430             }
431             
432             _numPorts = Math.ceil(pinCounter / 8);
433             debug("debug: Num ports = " + _numPorts);
434             
435             // Initialize port values
436             for (var j = 0; j < _numPorts; j++) {
437                 _digitalPort[j] = 0;
438             }
439             
440             _totalPins = pinCounter;
441             debug("debug: Num pins = " + _totalPins);
442 
443             // Map the analog pins to the board pins
444             // This will map the IOBoard analog pin numbers (printed on IOBoard)
445             // to their digital pin number equivalents
446             queryAnalogMapping();
447         }

449         /**
450          * Map map analog pins to board pin numbers. Need to do this because
451          * the capability query does not provide the correct order of analog
452          * pins.
453          *
454          * @private
455          */
456         function processAnalogMappingResponse(msg) {
457             // If running in multi-client mode and this client is 
458             // already configured ignore analog mapping response
459             if (_isConfigured) {
460                 return;
461             }
462 
463             var len = msg.length;
464             for (var i = 1; i < len; i++) {
465                 if (msg[i] != 127) {
466                     _analogPinMapping[msg[i]] = i - 1;
467                     _self.getPin(i - 1).setAnalogNumber(msg[i]);
468                 }
469             }
470             
471             if (!_isMultiClientEnabled) {
472                 startupInSingleClientMode();
473             } else {
474                 startupInMultiClientMode();
475             }
476         }

496          * Single client mode is the default mode.
497          * Checking the "Enable multi-client" box in the Breakout Server UI to
498          * enable multi-client mode.

508

510             setTimeout(startup, 500);
511             _isConfigured = true;

534         /**
535          * Reads the current configuration of the requested pin. The following
536          * values are returned: 1: pin number, 2: pin type (0: DIN, 1: DOUT, 
537          * 2: AIN, 3: AOUT / PWM, 4: SERVO, 5: SHIFT, 6: I2C), 3: pin state.
538          * The pin state for output modes is the value previously written
539          * to the pin. For input modes (AIN, DIN, etc) the state is typically
540          * zero (it is not the value that was written to the pin). For digital
541          * inputs the state is the status of the pullup resistor.
542          *
543          * @private
544          */
545         function processPinStateResponse(msg) {
546             // Ignore requests that were not made by this client
547             if (!_pinStateRequested) {
548                 return;
549             }
550                         
551             var len = msg.length,
552                 pinNumber = msg[1],
553                 pinType = msg[2],
554                 pinState,
555                 pin = _ioPins[pinNumber];
556 
557             if (len > 4) {
558                 pinState = _self.getValueFromTwo7bitBytes(msg[3], msg[4]);
559             } else if (len > 3) {
560                 pinState = msg[3];
561             }
562             
563             // update the pin type if it has changed
564             // typically this only happens when multiple clients are connecting
565             // to a single IOBoard. Each client (aside from the initial client) 
566             // needs to get the current pin type
567             if (pin.getType() != pinType) {
568                 pin.setType(pinType);
569                 managePinListener(pin);
570             }
571             
572             _pinStateRequested = false;
573             _self.dispatchEvent(new IOBoardEvent(IOBoardEvent.PIN_STATE_RESPONSE), {pin: pin});
574         }

587         /**
588          * Called when ever a pin value is set via pin.value = someValue.
589          * Sends digital or analog output pin and output values to the IOBoard.
590          *
591          * @private
592          * @param {Event} event A reference to the event object (Pin in this
593          * case).
594          */
595         function sendOut(event) {
596             var type = event.target.getType(),
597                 pinNum = event.target.number,
598                 value = event.target.value;
599 
600             switch (type) {
601             case Pin.DOUT:
602                 sendDigitalData(pinNum, value);
603                 break;
604             case Pin.AOUT:
605                 sendAnalogData(pinNum, value);
606                 break;
607             case Pin.SERVO:
608                 sendServoData(pinNum, value);
609                 break;
610             }
611         }

613         /**
614          * Ensure that event listeners are properly managed for pin objects 
615          * as the pin type is changed during the execution of the program.
616          *
617          * @private
618          */  
619         function managePinListener(pin) {
620             if (pin.getType() == Pin.DOUT || pin.getType() == Pin.AOUT || pin.getType() == Pin.SERVO) {
621                 if (!pin.hasEventListener(PinEvent.CHANGE)) {
622                     pin.addEventListener(PinEvent.CHANGE, sendOut);
623                 }
624             } else {
625                 if (pin.hasEventListener(PinEvent.CHANGE)) {
626                     try {
627                         pin.removeEventListener(PinEvent.CHANGE, sendOut);
628                     } catch (e) {
629                         // Pin had reference to other handler, ignore
630                         debug("debug: Caught pin removeEventListener exception");
631                     }
632                 }
633             }
634         }

636         /**
637          * @private
638          */
639         function sendAnalogData(pin, value) {
640 
641             var pwmMax = _self.getDigitalPin(pin).maxPWMValue;
642             value *= pwmMax;
643             value = (value < 0) ? 0: value;
644             value = (value > pwmMax) ? pwmMax : value;
645 
646             if (pin > 15 || value > Math.pow(2, 14)) {
647                 sendExtendedAnalogData(pin, value);
648             } else {
649                 _self.send([ANALOG_MESSAGE | (pin & 0x0F), value & 0x007F, (value >> 7) & 0x007F]);
650             }
651         }
652 
653         /**
654          * @private
655          */ 
656         function sendExtendedAnalogData(pin, value) {
657             var analogData = [];
658             
659             // If > 16 bits
660             if (value > Math.pow(2, 16)) {
661                 var err = "error: Extended Analog values > 16 bits are not currently supported by StandardFirmata";
662                 console.log(err);
663                 throw err;
664             }
665             
666             analogData[0] = START_SYSEX;
667             analogData[1] = EXTENDED_ANALOG;
668             analogData[2] = pin;
669             analogData[3] = value & 0x007F;
670             analogData[4] = (value >> 7) & 0x007F;  // Up to 14 bits
671                     
672             // If > 14 bits
673             if (value >= Math.pow(2, 14)) {
674                 analogData[5] = (value >> 14) & 0x007F;
675             }
676 
677             analogData.push(END_SYSEX);
678             _self.send(analogData);
679         }

681         /**
682          * @private
683          */
684         function sendDigitalData(pin, value) {
685             var portNum = Math.floor(pin / 8);
686 
687             if (value == Pin.HIGH) {
688                 // Set the bit
689                 _digitalPort[portNum] |= (value << (pin % 8));
690             }
691             else if (value == Pin.LOW) {
692                 // Clear the bit
693                 _digitalPort[portNum] &= ~(1 << (pin % 8));
694             }
695             else {
696                 console.log("warning: Invalid value passed to sendDigital, value must be 0 or 1.");
697                 return; // Invalid value
698             }
699             
700             _self.sendDigitalPort(portNum, _digitalPort[portNum]);  
701         }

703         /**
704          * @private
705          */ 
706         function sendServoData(pin, value) {
707             var servoPin = _self.getDigitalPin(pin);
708             if (servoPin.getType() == Pin.SERVO && servoPin.lastValue != value) {
709                 sendAnalogData(pin, value);
710             }   
711         }

731         /**
732          * Call this method to enable or disable analog input for the specified
733          * pin.
734          *
735          * @private
736          * @param {Number} pin The pin connected to the analog input
737          * @param {Number} mode Pin.ON to enable input or Pin.OFF to disable
738          * input for the specified pin.

1001         /**
1002          * Send a string message to the IOBoard. This is useful if you have a
1003          * custom sketch running on the IOBoard rather than StandardFirmata
1004          * and want to communicate with your javascript message via string
1005          * messages that you then parse in javascript.
1006          * You can receive string messages as well.
1007          *
1008          * <p>To test, load the EchoString.pde example from Firmata->Examples
1009          * menu in the IOBoard Application, then use sendString("your string
1010          * message") to have it echoed back to your javascript application.</p>
1011          * 
1012          * @param {String} str The string message to send to the IOBoard
1013          */
1014         this.sendString = function (str) {
1015             // Convert chars to decimal values
1016             var decValues = [];
1017             for (var i = 0, len = str.length; i < len; i++) {
1018                 decValues.push(toDec(str[i]) & 0x007F);
1019                 decValues.push((toDec(str[i]) >> 7) & 0x007F);
1020             }
1021             // Data > 7 bits in length must be split into 2 bytes and  
1022             // packed into an array before passing to the sendSysex
1023             // method
1024             this.sendSysex(STRING_DATA, decValues);
1025         };

1031          * calling it from your main application.
1032          *
1033          * @private
1034          * @param {Number} command The sysEx command value (see firmata.org)
1035          * @param {Number[]} data A packet of data representing the sysEx
1036          * message to be sent
1037          * @see <a href="http://firmata.org/wiki/Protocol#Sysex_Message_Format">Firmata Sysex Message Format"</a>
1038          */
1039         this.sendSysex = function (command, data) {
1040             var sysexData = [];
1041             sysexData[0] = START_SYSEX;
1042             sysexData[1] = command;
1043             // This would be problematic since the sysEx message format does
1044             // not enforce splitting all bytes after the command byte
1045             //for (var i=0, len=data.length; i<len; i++) {
1046             //  sysexData.push(data[i] & 0x007F);
1047             //  sysexData.push((data[i] >> 7) & 0x007F);                
1048             //}
1049             
1050             for (var i = 0, len = data.length; i < len; i++) {

1061          * pulse width. If you can't find it, then the default values should

1064          *
1065          * @param {Number} pin The pin the server is connected to.
1066          * @param {Number} minPulse [optional] The minimum pulse width for the
1067          * servo. Default = 544.
1068          * @param {Number} maxPulse [optional] The maximum pulse width for the

1071         this.sendServoAttach = function (pin, minPulse, maxPulse) {
1072             var servoPin,
1073                 servoData = [];
