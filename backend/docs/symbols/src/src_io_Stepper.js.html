  6 JSUTILS.namespace('BO.io.Stepper');
  7 
  8 BO.io.Stepper = (function () {
  9     "use strict";
 10 
 11     var Stepper;
 12     var instanceCounter = 0;
 13 
 14     // private static constants
 15     var STEPPER = 0x72,
 16         CONFIG = 0,
 17         STEP = 1,
 18         MAX_STEPS = 2097151, // 21 bits (2^21 - 1)
 19         MAX_SPEED = 16383; // 14 bits (2^14 - 1)
 20 
 21     // dependencies
 22     var Pin = BO.Pin,
 23         EventDispatcher = JSUTILS.EventDispatcher,
 24         Event = JSUTILS.Event,
 25         IOBoardEvent = BO.IOBoardEvent;
 26 
 27     /**
 28      * Creates a new Stepper.
 29      *
 30      * @exports Stepper as BO.io.Stepper
 31      * @class Creates an interface to a Stepper motor. Use this object to set
 32      * the direction and number of steps for the motor to rotate. See
 33      * Breakout/examples/actuators/stepper_2wire.html, stepper_4wire.html, 
 34      * stepper_easydriver.html and stepper_simple.html for example applications.
 35      *
 36      * @constructor
 37      * @param {IOBoard} board A reference to the IOBoard instance that the 
 38      * stepper is attached to.
 39      * @param {Number} driverType. The type of driver (Stepper.DRIVER, 
 40      * Stepper.TWO_WIRE, or Stepper.FOUR_WIRE).
 41      * @param {Number} numStepsPerRev The number of steps to make 1 revolution. 
 42      * @param {Pin} directionPin If dirver interface, the pin used to control 
 43      * the direction.
 44      * If 2-wire or 4-wire interface, the 1st moter pin.
 45      * @param {Pin} stepPin If dirver interface, the pin used to control the 
 46      * steps.
 47      * If 2-wire or 4-wire interface, the 2nd moter pin.
 48      * @param {Pin} motorPin3 [optional] Only required for a 4-wire interface.
 49      * @param {Pin} motorPin4 [optional] Only required for a 4-wire interface.
 50      *
 51      * @example
 52      * // example
 53      * var Stepper = BO.io.Stepper,
 54      *     Event = JSUTILS.Event;
 55      *
 56      * var stepper,
 57      *     stepsPerRev = 200,           // update this for your stepper
 58      *     numSteps = stepsPerRev * 10, // 10 revolutions (+ CW, - CCW)
 59      *     speed = 15.0,                // rad/sec (RPM = speed * 9.55)
 60      *     acceleration = 20.0,         // rad/sec^2
 61      *     deceleration = 20.0;         // rad/sec^2
 62      *
 63      * stepper = new Stepper(arduino,
 64      *              Stepper.TWO_WIRE, // or Stepper.DRIVER or Stepper.FOUR_WIRE
 65      *              stepsPerRev,
 66      *              arduino.getDigitalPin(2),
 67      *              arduino.getDigitalPin(3));
 68      *
 69      * stepper.addEventListener(Event.COMPLETE, onStepperComplete);
 70      *
 71      * // acceleration and deceleration parameters are optional
 72      * stepper.step(numSteps, speed, acceleration, deceleration);
 73      *
 74      * function onStepperComplete(event) {
 75      *     // each stepper is assigned a read-only id value when instantiated
 76      *     console.log("stepper " + event.target.id + " sequence complete");
 77      * }
 78      */
 79     Stepper = function (board, driverType, numStepsPerRev, directionPin, stepPin, motorPin3, motorPin4) {
 80         
 81         // create a new id each time a new instance is created
 82         this._id = instanceCounter++;
 83         if (this._id > 5) {
 84             console.log("Warning: A maximum of 6 Stepper instances can be created");
 85             return;
 86         }
 87 
 88         this.name = "Stepper";
 89         this._board = board;
 90         this._evtDispatcher = new EventDispatcher(this);
 91 
 92         var numStepsPerRevLSB = numStepsPerRev & 0x007F,
 93             numStepsPerRevMSB = (numStepsPerRev >> 7) & 0x007F;
 94 
 95         this._board.setDigitalPinMode(directionPin.number, Pin.DOUT);
 96         this._board.setDigitalPinMode(stepPin.number, Pin.DOUT);
 97 
 98         this._board.addEventListener(IOBoardEvent.SYSEX_MESSAGE, this.onSysExMessage.bind(this));
 99         
100         switch (driverType) {
101         case Stepper.DRIVER:
102         case Stepper.TWO_WIRE:
103             // configure the stepper motor
104             this._board.sendSysex(STEPPER,
105                                     [CONFIG,
106                                     this._id,
107                                     driverType,
108                                     numStepsPerRevLSB, 
109                                     numStepsPerRevMSB,
110                                     directionPin.number,
111                                     stepPin.number]);
112             break;
113         case Stepper.FOUR_WIRE:
114             this._board.setDigitalPinMode(motorPin3.number, Pin.DOUT);
115             this._board.setDigitalPinMode(motorPin4.number, Pin.DOUT);
116 
117             // configure the stepper motor
118             this._board.sendSysex(STEPPER,
119                                     [CONFIG,
120                                     this._id,
121                                     driverType,
122                                     numStepsPerRevLSB, 
123                                     numStepsPerRevMSB,
124                                     directionPin.number,
125                                     stepPin.number,
126                                     motorPin3.number,
127                                     motorPin4.number]);
128             break;
129         }
130 
131     };
132 
133     Stepper.prototype = {
134 
135         /**
136          * Move the stepper a given number of steps at the specified
137          * speed (rad/sec), acceleration (rad/sec^2) and deceleration (rad/sec^2).
138          * The accel and decel parameters are optional but if using, both values
139          * must be passed to the function.
140          *
141          * @param {Number} numSteps The number ofsteps to move the motor (max = +/-2097151 (21 bits)).
142          * Positive value is clockwise, negative value is counter clockwise.
143          * @param {Number} speed Max speed in rad/sec (1 rad/sec = 9.549 RPM)
144          * (max precision of 2 decimal places)
145          * @param {Number} accel [optional] Acceleration in rad/sec^2 (max precision of 2 decimal places)
146          * @param {Number} decel [optional] Deceleration in rad/sec^2 (max precision of 2 decimal places)
147          */      
148         step: function (numSteps, speed, accel, decel) {
149             var steps,
150                 speedLSB,
151                 speedMSB,
152                 accelLSB,
153                 accelMSB,
154                 decelLSB,
155                 decelMSB,
156                 direction = Stepper.CLOCKWISE;
157 
158             if (numSteps > MAX_STEPS) {
159                 numSteps = MAX_STEPS;
160                 console.log("Warning: Maximum number of steps (2097151) exceeded. Setting to step number to 2097151");
161             }
162             if (numSteps < -MAX_STEPS) {
163                 numSteps = -MAX_STEPS;
164                 console.log("Warning: Maximum number of steps (-2097151) exceeded. Setting to step number to -2097151");
165             }
166 
167             if (numSteps > 0) {
168                 direction = Stepper.COUNTER_CLOCKWISE;
169             }
170 
171             steps = [
172                 Math.abs(numSteps) & 0x0000007F,
173                 (Math.abs(numSteps) >> 7) & 0x0000007F,
174                 (Math.abs(numSteps) >> 14) & 0x0000007F
175             ];
176 
177             // the stepper interface expects decimal expressed an an integer
178             speed = Math.floor(speed.toFixed(2) * 100);
179 
180             if (speed > MAX_SPEED) {
181                 speed = MAX_SPEED;
182                 console.log("Warning: Maximum speed (163.83 rad/sec) exceeded. Setting speed to 163.83 rad/sec");
183             }                   
184 
185             speedLSB = speed & 0x007F;
186             speedMSB = (speed >> 7) & 0x007F;
187 
188             // make sure both accel and decel are defined
189             if (accel !== undefined && decel !== undefined) {
190                 // the stepper interface expects decimal expressed an an integer
191                 accel = Math.floor(accel.toFixed(2) * 100);
192                 decel = Math.floor(decel.toFixed(2) * 100);
193 
194                 accelLSB = accel & 0x007F;
195                 accelMSB = (accel >> 7) & 0x007F;
196 
197                 decelLSB = decel & 0x007F;
198                 decelMSB = (decel >> 7) & 0x007F;               
199                             
200                 this._board.sendSysex(STEPPER, 
201                                         [STEP,
202                                         this._id,
203                                         direction,                              
204                                         steps[0],
205                                         steps[1],
206                                         steps[2],
207                                         speedLSB,
208                                         speedMSB,
209                                         accelLSB,
210                                         accelMSB,
211                                         decelLSB,
212                                         decelMSB
213                                         ]);
214             } else {
215                 // don't send accel and decel values
216                 this._board.sendSysex(STEPPER, 
217                                         [STEP,
218                                         this._id,
219                                         direction,                                  
220                                         steps[0],
221                                         steps[1],
222                                         steps[2],
223                                         speedLSB,
224                                         speedMSB
225                                         ]);             
226             }
227         },
228 
229         /**
230          * Listen for stepping complete event
231          *
232          * @private
233          */
234         onSysExMessage: function (event) {
235             var message = event.message;
236 
237             if (message[0] !== STEPPER) {
238                 return;
239             } else if (message[1] !== this._id) {
240                 return;
241             } else {
242                 this.dispatchEvent(new Event(Event.COMPLETE));
243             }
244         },
245 
246         /**
247          * @return {Number} The id of the Stepper object instance
248          */
249         get id() {
250             return this._id;
251         },
252 
253         /* implement EventDispatcher */
254         
255         /**
256          * @param {String} type The event type
257          * @param {Function} listener The function to be called when the event is fired
258          */
259         addEventListener: function (type, listener) {
260             this._evtDispatcher.addEventListener(type, listener);
261         },
262         
263         /**
264          * @param {String} type The event type
265          * @param {Function} listener The function to be called when the event is fired
266          */
267         removeEventListener: function (type, listener) {
268             this._evtDispatcher.removeEventListener(type, listener);
269         },
270         
271         /**
272          * @param {String} type The event type
273          * return {boolean} True is listener exists for this type, false if not.
274          */
275         hasEventListener: function (type) {
276             return this._evtDispatcher.hasEventListener(type);
277         },
278         
279         /**
280          * @param {Event} type The Event object
281          * @param {Object} optionalParams Optional parameters to assign to the event object.
282          * return {boolean} True if dispatch is successful, false if not.
283          */     
284         dispatchEvent: function (event, optionalParams) {
285             return this._evtDispatcher.dispatchEvent(event, optionalParams);
286         }           
287     
288     };
289 
290     /** @constant */
291     Stepper.CLOCKWISE = 0;
292     /** @constant */
293     Stepper.COUNTER_CLOCKWISE = 1;
294     /** @constant */
295     Stepper.DRIVER = 1;
296     /** @constant */
297     Stepper.TWO_WIRE = 2;
298     /** @constant */
299     Stepper.FOUR_WIRE = 4;              
300 
301     return Stepper;
302 
303 })();
