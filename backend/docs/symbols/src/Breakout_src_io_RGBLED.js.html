 12 JSUTILS.namespace('BO.io.RGBLED');
 13 
 14 BO.io.RGBLED = (function() {
 15 
 16 	var RGBLED;
 17 
 18 	// Dependencies
 19 	var Pin = BO.Pin,
 20 		LED = BO.io.LED;
 21 
 22 	/**
 23 	 * The RGB pins of the RGBLED must be connected to PWM pins on the IOBoard.
 24 	 *
 25 	 * <p>COMMON_ANODE vs COMMON_CATHODE. You can determine if your RGB LED is 
 26 	 * common anode or common cathode by reading the datasheet. To wire a 
 27 	 * common cathode RGB LED, connect the cathode to ground and the 3 anode
 28 	 * pins to the IOBoard PWM pins via 330 ohm resistors. For a common anode
 29 	 * LED, the anode is connected to power and the 3 cathode pins are connected
 30 	 * to the IOBoard PWM pins via 330 ohm resistors.</p>
 31 	 *
 32 	 * @exports RGBLED as BO.io.RGBLED
 33 	 * @class Creates an interface to an RGB LED. This interface is for the
 34 	 * type of RGB LED with 4 legs. One leg is connected to power or ground 
 35 	 * (depending on the type of LED - common anode or common cathode) and the
 36 	 * other 3 legs are connected to PWM pins on the I/O board. See 
 37 	 * Breakout/examples/schematics.pdf for wiring diagrams. See 
 38 	 * Breakout/examples/actuators/rgb_led.html for an example application.
 39 	 * @constructor
 40 	 * @param {IOBoard} board A reference to the IOBoard instance that the
 41 	 * servo is attached to.
 42 	 * @param {Pin} redLEDPin A reference to the IOBoard Pin the red LED pin
 43 	 * is connected to.
 44 	 * @param {Pin} greenLEDPin A reference to the IOBoard Pin the green LED
 45 	 * pin is connected to.
 46 	 * @param {Pin} blueLEDPin A reference to the IOBoard Pin the blue LED pin
 47 	 * is connected to.	 
 48 	 * @param {Number} driveMode The drive mode of the RGBLED. Must be set to
 49 	 * RGBLED.COMMON_ANODE or RGBLED.COMMON_CATHODE. RGBLED.COMMON_ANODE is
 50 	 * default.
 51 	 */
 52 	RGBLED = function(board, redLEDPin, greenLEDPin, blueLEDPin, driveMode) {
 53 		"use strict";
 54 		
 55 		this.name = "RGBLED";
 56 
 57 		if (driveMode === undefined) driveMode = RGBLED.COMMON_ANODE;
 58 
 59 		this._redLED = new LED(board, redLEDPin, driveMode);
 60 		this._greenLED = new LED(board, greenLEDPin, driveMode);
 61 		this._blueLED = new LED(board, blueLEDPin, driveMode);
 62 	};
 63 
 64 	RGBLED.prototype = {
 65 
 66 		/**
 67 		 * Set the RGBLED color.
 68 		 * 
 69 		 * @param {Number} red The red value (0 - 255)
 70 		 * @param {Number} green The green value (0 - 255)
 71 		 * @param {Number} blue The blue value (0 - 255)
 72 		 */
 73 		setColor: function(red, green, blue) {
 74 			red = red / 255;
 75 			green = green / 255;
 76 			blue = blue / 255;
 77 
 78 			this._redLED.intensity = red;
 79 			this._greenLED.intensity = green;
 80 			this._blueLED.intensity = blue;
 81 		},
 82 
 83 		/**
 84 		 * Fade in the RGBLED from the off state.
 85 		 * 
 86 		 * @param {Number} time The time of the fade (in milliseconds)
 87 		 */
 88 		fadeIn: function(time) {
 89 			time = time || 1000;
 90 			this._redLED.fadeTo(1, time);
 91 			this._greenLED.fadeTo(1, time);
 92 			this._blueLED.fadeTo(1, time);
 93 		},
 94 
 95 		/**
 96 		 * Fade out the RGBLED from the on state.
 97 		 * 
 98 		 * @param {Number} time The time of the fade (in milliseconds)
 99 		 */
100 		fadeOut: function(time) {
101 			time = time || 1000;
102 			this._redLED.fadeTo(0, time);
103 			this._greenLED.fadeTo(0, time);
104 			this._blueLED.fadeTo(0, time);			
105 		},
106 
107 		/**
108 		 * Fade from the current color to the new color.
109 		 * 
110 		 * @param {Number} red The red value to fade to (0 - 255)
111 		 * @param {Number} green The green value to fade to (0 - 255)
112 		 * @param {Number} blue The blue value to fade to (0 - 255)
113 		 * @param {Number} time The time of the fade (in milliseconds)		 
114 		 */		
115 		fadeTo: function(red, green, blue, time) {
116 			red = red / 255;
117 			green = green / 255;
118 			blue = blue / 255;
119 			time = time || 1000;
120 
121 			this._redLED.fadeTo(red, time);
122 			this._greenLED.fadeTo(green, time);
123 			this._blueLED.fadeTo(blue, time);
124 		}
125 	};
126 
127 	/** @constant */
128 	RGBLED.COMMON_ANODE = LED.SYNC_DRIVE;
129 	/** @constant */
130 	RGBLED.COMMON_CATHODE = LED.SOURCE_DRIVE;				
131 
132 	return RGBLED;
133 
134 }());
